238    /**
239     * Converts a pixel's X coordinate into a horizontal tile index
240     * relative to a given tile grid layout specified by its X offset
241     * and tile width.
242     *
243     * <p> If <code>tileWidth < 0</code>, the results of this method
244     * are undefined.  If <code>tileWidth == 0</code>, an
245     * <code>ArithmeticException</code> will be thrown.
246     *
247     * @throws ArithmeticException  If <code>tileWidth == 0</code>.
248     */
249    public static int XToTileX(int x, int tileGridXOffset, int tileWidth) {
250        x -= tileGridXOffset;
251        if (x < 0) {
252            x += 1 - tileWidth;         // force round to -infinity (ceiling)
253        }
254        return x/tileWidth;
255    }

258     * Converts a pixel's Y coordinate into a vertical tile index
259     * relative to a given tile grid layout specified by its Y offset
260     * and tile height.
261     *
262     * <p> If <code>tileHeight < 0</code>, the results of this method
263     * are undefined.  If <code>tileHeight == 0</code>, an
264     * <code>ArithmeticException</code> will be thrown.
265     *
266     * @throws ArithmeticException  If <code>tileHeight == 0</code>.

269        y -= tileGridYOffset;
270        if (y < 0) {
271            y += 1 - tileHeight;         // force round to -infinity (ceiling)
272        }
273        return y/tileHeight;
274    }

284    public void setOutput(Object output) {
285        super.setOutput(output);
286
287        if (output != null) {
288            if (!(output instanceof ImageOutputStream)) {
289                throw new IllegalArgumentException
290                    ("output not an ImageOutputStream!");
291            }
292            this.stream = (ImageOutputStream)output;
293
294            //
295            // The output is expected to be positioned at a TIFF header
296            // or at some arbitrary location which may or may not be
297            // the EOF. In the former case the writer should be able
298            // either to overwrite the existing sequence or append to it.
299            //
300
301            // Set the position of the header and the next available space.
302            try {
303                headerPosition = this.stream.getStreamPosition();
304                try {
305                    // Read byte order and magic number.
306                    byte[] b = new byte[4];
307                    stream.readFully(b);
308
309                    // Check bytes for TIFF header.
310                    if((b[0] == (byte)0x49 && b[1] == (byte)0x49 &&
311                        b[2] == (byte)0x2a && b[3] == (byte)0x00) ||
312                       (b[0] == (byte)0x4d && b[1] == (byte)0x4d &&
313                        b[2] == (byte)0x00 && b[3] == (byte)0x2a)) {
314                        // TIFF header.
315                        this.nextSpace = stream.length();
316                    } else {
317                        // Neither TIFF header nor EOF: overwrite.
318                        this.nextSpace = headerPosition;
319                    }
320                } catch(IOException io) { // thrown by readFully()
321                    // At EOF or not at a TIFF header.
322                    this.nextSpace = headerPosition;
323                }
324                stream.seek(headerPosition);
325            } catch(IOException ioe) { // thrown by getStreamPosition()
326                // Assume it's at zero.
327                this.nextSpace = headerPosition = 0L;
328            }
329        } else {
330            this.stream = null;
331        }
332    }

362                                             ImageWriteParam param) {
363        // Check arguments.
364        if(inData == null) {
365            throw new IllegalArgumentException("inData == null!");
366        }
367
368        // Note: param is irrelevant as it does not contain byte order.
369
370        TIFFStreamMetadata outData = null;
371        if(inData instanceof TIFFStreamMetadata) {
372            outData = new TIFFStreamMetadata();
373            outData.byteOrder = ((TIFFStreamMetadata)inData).byteOrder;
374            return outData;
375        } else if(Arrays.asList(inData.getMetadataFormatNames()).contains(
376                      TIFFStreamMetadata.nativeMetadataFormatName)) {
377            outData = new TIFFStreamMetadata();
378            String format = TIFFStreamMetadata.nativeMetadataFormatName;
379            try {
380                outData.mergeTree(format, inData.getAsTree(format));
381            } catch(IIOInvalidTreeException e) {
382                // XXX Warning
383            }
384        }
385
386        return outData;
387    }

389    public IIOMetadata
390        convertImageMetadata(IIOMetadata inData,
391                             ImageTypeSpecifier imageType,
392                             ImageWriteParam param) {
393        // Check arguments.
394        if(inData == null) {
395            throw new IllegalArgumentException("inData == null!");
396        }
397        if(imageType == null) {
398            throw new IllegalArgumentException("imageType == null!");
399        }
400
401        TIFFImageMetadata outData = null;
402
403        // Obtain a TIFFImageMetadata object.
404        if(inData instanceof TIFFImageMetadata) {
405            // Create a new metadata object from a clone of the input IFD.
406            TIFFIFD inIFD = ((TIFFImageMetadata)inData).getRootIFD();
407            outData = new TIFFImageMetadata(inIFD.getShallowClone());
408        } else if(Arrays.asList(inData.getMetadataFormatNames()).contains(
409                      TIFFImageMetadata.nativeMetadataFormatName)) {
410            // Initialize from the native metadata form of the input tree.
411            try {
412                outData = convertNativeImageMetadata(inData);
413            } catch(IIOInvalidTreeException e) {
414                // XXX Warning
415            }
416        } else if(inData.isStandardMetadataFormatSupported()) {
417            // Initialize from the standard metadata form of the input tree.
418            try {
419                outData = convertStandardImageMetadata(inData);
420            } catch(IIOInvalidTreeException e) {
421                // XXX Warning
422            }
423        }
424
425        // Update the metadata per the image type and param.
426        if(outData != null) {
427            TIFFImageWriter bogusWriter =
428                new TIFFImageWriter(this.originatingProvider);
429            bogusWriter.imageMetadata = outData;
430            bogusWriter.param = param;
431            SampleModel sm = imageType.getSampleModel();
432            try {
433                bogusWriter.setupMetadata(imageType.getColorModel(), sm,
434                                          sm.getWidth(), sm.getHeight());
435                return bogusWriter.imageMetadata;
436            } catch(IIOException e) {
437                // XXX Warning
438                return null;
439            }
440        }
441
442        return outData;
443    }

447     * <code>TIFFImageMetadata</code> object.
448     *
449     * @param inData The metadata object.
450     * @return a <code>TIFFImageMetadata</code> or <code>null</code> if

454     * @throws IIOInvalidTreeException if <code>inData</code> generates an
455     * invalid standard metadata tree.

458        throws IIOInvalidTreeException {
459
460        if(inData == null) {
461            throw new IllegalArgumentException("inData == null!");
462        } else if(!inData.isStandardMetadataFormatSupported()) {
463            throw new IllegalArgumentException
464                ("inData does not support standard metadata format!");
465        }
466
467        TIFFImageMetadata outData = null;
468
469        String formatName = IIOMetadataFormatImpl.standardMetadataFormatName;
470        Node tree = inData.getAsTree(formatName);
471        if (tree != null) {
472            List tagSets = new ArrayList(1);
473            tagSets.add(BaselineTIFFTagSet.getInstance());
474            outData = new TIFFImageMetadata(tagSets);
475            outData.setFromTree(formatName, tree);
476        }
477
478        return outData;
479    }

519    /**
520     * Sets up the output metadata adding, removing, and overriding fields
521     * as needed. The destination image dimensions are provided as parameters
522     * because these might differ from those of the source due to subsampling.
523     * 
524     * @param cm The <code>ColorModel</code> of the image being written.
525     * @param sm The <code>SampleModel</code> of the image being written.
526     * @param destWidth The width of the written image after subsampling.
527     * @param destHeight The height of the written image after subsampling.
528     */
529    void setupMetadata(ColorModel cm, SampleModel sm,
530                       int destWidth, int destHeight)
531        throws IIOException {
532        // Get initial IFD from metadata
533
534        // Always emit these fields:
535        //
536        // Override values from metadata:
537        //
538        //  planarConfiguration -> chunky (planar not supported on output)
539        //
540        // Override values from metadata with image-derived values:
541        // 
542        //  bitsPerSample (if not bilivel)
543        //  colorMap (if palette color)
544        //  photometricInterpretation (derive from image)
545        //  imageLength
546        //  imageWidth
547        // 
548        //  rowsPerStrip     \      /   tileLength
549        //  stripOffsets      | OR |   tileOffsets
550        //  stripByteCounts  /     |   tileByteCounts
551        //                          \   tileWidth
552        //                    
553        //
554        // Override values from metadata with write param values:
555        //
556        //  compression
557
558        // Use values from metadata if present for these fields,
559        // otherwise use defaults:
560        //
561        //  resolutionUnit
562        //  XResolution (take from metadata if present)
563        //  YResolution
564        //  rowsPerStrip
565        //  sampleFormat
566
567        TIFFIFD rootIFD = imageMetadata.getRootIFD();
568
569        BaselineTIFFTagSet base = BaselineTIFFTagSet.getInstance();
570
571        // If PlanarConfiguration field present, set value to chunky.
572
573        TIFFField f =
574            rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_PLANAR_CONFIGURATION);
575        if(f != null &&
576           f.getAsInt(0) != BaselineTIFFTagSet.PLANAR_CONFIGURATION_CHUNKY) {
577            // XXX processWarningOccurred()
578            TIFFField planarConfigurationField =
579                new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_PLANAR_CONFIGURATION),
580                              BaselineTIFFTagSet.PLANAR_CONFIGURATION_CHUNKY);
581            rootIFD.addTIFFField(planarConfigurationField);
582        }
583
584        char[] extraSamples = null;
585
586        this.photometricInterpretation = -1;
587        boolean forcePhotometricInterpretation = false;
588
589        f =
590       rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_PHOTOMETRIC_INTERPRETATION);
591        if (f != null) {
592            photometricInterpretation = f.getAsInt(0);
593            if(photometricInterpretation ==
594               BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_PALETTE_COLOR &&
595               !(cm instanceof IndexColorModel)) {
596                photometricInterpretation = -1;
597            } else {
598                forcePhotometricInterpretation = true;
599            }
600        }
601
602//         f = rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_EXTRA_SAMPLES);
603//         if (f != null) {
604//             extraSamples = f.getAsChars();
605//         }
606
607//         f = rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_BITS_PER_SAMPLE);
608//         if (f != null) {
609//             bitsPerSample = f.getAsChars();
610//         }
611
612        int[] sampleSize = sm.getSampleSize();
613
614        int numBands = sm.getNumBands();
615        int numExtraSamples = 0;
616
617        // Check that numBands > 1 here because TIFF requires that
618        // SamplesPerPixel = numBands + numExtraSamples and numBands
619        // cannot be zero.
620        if (numBands > 1 && cm != null && cm.hasAlpha()) {
621            --numBands;
622            numExtraSamples = 1;
623            extraSamples = new char[1];
624            if (cm.isAlphaPremultiplied()) {
625                extraSamples[0] =
626                    BaselineTIFFTagSet.EXTRA_SAMPLES_ASSOCIATED_ALPHA;
627            } else {
628                extraSamples[0] =
629                    BaselineTIFFTagSet.EXTRA_SAMPLES_UNASSOCIATED_ALPHA;
630            }
631        }
632
633        if (numBands == 3) {
634            this.nativePhotometricInterpretation =
635                BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_RGB;
636            if (photometricInterpretation == -1) {
637                photometricInterpretation =
638                    BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_RGB;
639            }
640        } else if (sm.getNumBands() == 1 && cm instanceof IndexColorModel) {
641            IndexColorModel icm = (IndexColorModel)cm;
642            int r0 = icm.getRed(0);
643            int r1 = icm.getRed(1);
644            if (icm.getMapSize() == 2 &&
645                (r0 == icm.getGreen(0)) && (r0 == icm.getBlue(0)) &&
646                (r1 == icm.getGreen(1)) && (r1 == icm.getBlue(1)) &&
647                (r0 == 0 || r0 == 255) &&
648                (r1 == 0 || r1 == 255) &&
649                (r0 != r1)) {
650                // Black/white image
651
652                if (r0 == 0) {
653                    nativePhotometricInterpretation =
654                   BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_BLACK_IS_ZERO;
655                } else {
656                    nativePhotometricInterpretation =
657                   BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_WHITE_IS_ZERO;
658                }
659
660
661                // If photometricInterpretation is already set to
662                // WhiteIsZero or BlackIsZero, leave it alone
663                if (photometricInterpretation !=
664                 BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_BLACK_IS_ZERO &&
665                    photometricInterpretation !=
666                 BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_WHITE_IS_ZERO) {
667                    photometricInterpretation =
668                        r0 == 0 ?
669                  BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_BLACK_IS_ZERO :
670                  BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_WHITE_IS_ZERO;
671                }
672            } else {
673                nativePhotometricInterpretation =
674                photometricInterpretation =
675                   BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_PALETTE_COLOR;
676            }
677        } else {
678            if(cm != null) {
679                switch(cm.getColorSpace().getType()) {
680                case ColorSpace.TYPE_Lab:
681                    nativePhotometricInterpretation =
682                        BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_CIELAB;
683                    break;
684                case ColorSpace.TYPE_YCbCr:
685                    nativePhotometricInterpretation =
686                        BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_Y_CB_CR;
687                    break;
688                case ColorSpace.TYPE_CMYK:
689                    nativePhotometricInterpretation =
690                        BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_CMYK;
691                    break;
692                default:
693                    nativePhotometricInterpretation =
694                        BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_BLACK_IS_ZERO;
695                }
696            } else {
697                nativePhotometricInterpretation =
698                    BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_BLACK_IS_ZERO;
699            }
700            if (photometricInterpretation == -1) {
701                photometricInterpretation = nativePhotometricInterpretation;
702            }
703        }
704
705        // Set the compressor and color converter.
706
707        this.compressor = null;
708        this.colorConverter = null;
709        if (param instanceof TIFFImageWriteParam) {
710            TIFFImageWriteParam tparam = (TIFFImageWriteParam)param;
711            if(tparam.getCompressionMode() == tparam.MODE_EXPLICIT) {
712                compressor = tparam.getTIFFCompressor();
713                String compressionType = param.getCompressionType();
714                if(compressor != null &&
715                   !compressor.getCompressionType().equals(compressionType)) {
716                    // Unset the TIFFCompressor if its compression type is
717                    // not the one selected.
718                    compressor = null;
719                }
720            } else {
721                // Compression mode not MODE_EXPLICIT.
722                compressor = null;
723            }
724            colorConverter = tparam.getColorConverter();
725            if (colorConverter != null) {
726                photometricInterpretation =
727                    tparam.getPhotometricInterpretation();
728            }
729        }
730
731        // Emit compression tag
732
733        int compressionMode = param instanceof TIFFImageWriteParam ?
734            param.getCompressionMode() : ImageWriteParam.MODE_DEFAULT;
735        switch(compressionMode) {
736        case ImageWriteParam.MODE_EXPLICIT:
737            {
738                String compressionType = param.getCompressionType();
739                if (compressionType == null) {
740                    this.compression = BaselineTIFFTagSet.COMPRESSION_NONE;
741                } else {
742                    // Determine corresponding compression tag value.
743                    int len = compressionTypes.length;
744                    for (int i = 0; i < len; i++) {
745                        if (compressionType.equals(compressionTypes[i])) {
746                            this.compression = compressionNumbers[i];
747                        }
748                    }
749                }
750
751                // Ensure the compressor, if any, matches compression setting
752                // with the precedence described in TIFFImageWriteParam.
753                if(compressor != null &&
754                   compressor.getCompressionTagValue() != this.compression) {
755                    // Does not match: unset the compressor.
756                    compressor = null;
757                }
758            }
759            break;
760        case ImageWriteParam.MODE_COPY_FROM_METADATA:
761            {
762                TIFFField compField =
763                    rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_COMPRESSION);
764                if(compField != null) {
765                    this.compression = compField.getAsInt(0);
766                    break;
767                }
768            }
769        case ImageWriteParam.MODE_DEFAULT:
770        case ImageWriteParam.MODE_DISABLED:
771        default:
772            this.compression = BaselineTIFFTagSet.COMPRESSION_NONE;
773        }
774
775        TIFFField predictorField =
776            rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_PREDICTOR);
777        if (predictorField != null) {
778            this.predictor = predictorField.getAsInt(0);
779            
780            // We only support Horizontal Predictor for a bitDepth of 8
781            if (sampleSize[0] != 8 || 
782                // Check the value of the tag for validity
783                (predictor != BaselineTIFFTagSet.PREDICTOR_NONE && 
784                 predictor != 
785                 BaselineTIFFTagSet.PREDICTOR_HORIZONTAL_DIFFERENCING)) {
786                // XXX processWarningOccured ???
787                // Set to default
788                predictor = BaselineTIFFTagSet.PREDICTOR_NONE;          
789
790                // Emit this changed predictor value to metadata
791                TIFFField newPredictorField =
792                   new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_PREDICTOR),
793                                 predictor);
794                rootIFD.addTIFFField(newPredictorField);
795            }
796
797            // XXX Do we need to ensure that predictor is not passed on if
798            // the compression is not either Deflate or LZW?
799        }
800
801        TIFFField compressionField =
802            new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_COMPRESSION),
803                          compression);
804        rootIFD.addTIFFField(compressionField);
805
806        // Set EXIF flag. Note that there is no way to determine definitively
807        // when an uncompressed thumbnail is being written as the EXIF IFD
808        // pointer field is optional for thumbnails.
809        boolean isEXIF = false;
810        if(numBands == 3 &&
811           sampleSize[0] == 8 && sampleSize[1] == 8 && sampleSize[2] == 8) {
812            // Three bands with 8 bits per sample.
813            if(rootIFD.getTIFFField(EXIFParentTIFFTagSet.TAG_EXIF_IFD_POINTER)
814               != null) {
815                // EXIF IFD pointer present.
816                if(compression == BaselineTIFFTagSet.COMPRESSION_NONE &&
817                   (photometricInterpretation ==
818                    BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_RGB ||
819                    photometricInterpretation ==
820                    BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_Y_CB_CR)) {
821                    // Uncompressed RGB or YCbCr.
822                    isEXIF = true;
823                } else if(compression ==
824                          BaselineTIFFTagSet.COMPRESSION_OLD_JPEG) {
825                    // Compressed.
826                    isEXIF = true;
827                }
828            } else if(compressionMode == ImageWriteParam.MODE_EXPLICIT &&
829                      EXIF_JPEG_COMPRESSION_TYPE.equals
830                      (param.getCompressionType())) {
831                // EXIF IFD pointer absent but EXIF JPEG compression set.
832                isEXIF = true;
833            }
834        }
835
836        // Initialize JPEG interchange format flag which is used to
837        // indicate that the image is stored as a single JPEG stream.
838        // This flag is separated from the 'isEXIF' flag in case JPEG
839        // interchange format is eventually supported for non-EXIF images.
840        boolean isJPEGInterchange =
841            isEXIF && compression == BaselineTIFFTagSet.COMPRESSION_OLD_JPEG;
842
843        if (compressor == null) {
844            if (compression == BaselineTIFFTagSet.COMPRESSION_CCITT_RLE) {
845
846                if(compressor == null) {
847                    compressor = new TIFFRLECompressor();
848                    if(DEBUG) {
849                        System.out.println("Using Java RLE compressor");
850                    }
851                }
852
853                if (!forcePhotometricInterpretation) {
854                    photometricInterpretation =
855                   BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_WHITE_IS_ZERO;
856                }
857            } else if (compression ==
858                       BaselineTIFFTagSet.COMPRESSION_CCITT_T_4) {
859                
860
861                if(compressor == null) {
862                    compressor = new TIFFT4Compressor();
863                    if(DEBUG) {
864                        System.out.println("Using Java T.4 compressor");
865                    }
866                }
867
868                if (!forcePhotometricInterpretation) {
869                    photometricInterpretation =
870                   BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_WHITE_IS_ZERO;
871                }
872            } else if (compression ==
873                       BaselineTIFFTagSet.COMPRESSION_CCITT_T_6) {
874               
875
876                if(compressor == null) {
877                    compressor = new TIFFT6Compressor();
878                    if(DEBUG) {
879                        System.out.println("Using Java T.6 compressor");
880                    }
881                }
882
883                if (!forcePhotometricInterpretation) {
884                    photometricInterpretation =
885                   BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_WHITE_IS_ZERO;
886                }
887            } else if (compression ==
888                       BaselineTIFFTagSet.COMPRESSION_LZW) {
889                compressor = new TIFFLZWCompressor(predictor);
890            } else if (compression ==
891                       BaselineTIFFTagSet.COMPRESSION_OLD_JPEG) {
892                if(isEXIF) {
893                    compressor = new TIFFEXIFJPEGCompressor(param);
894                } else {
895                    throw new IIOException
896                        ("Old JPEG compression not supported!");
897                }
898            } else if (compression ==
899                       BaselineTIFFTagSet.COMPRESSION_JPEG) {
900                if(numBands == 3 && sampleSize[0] == 8 &&
901                   sampleSize[1] == 8 && sampleSize[2] == 8) {
902                    photometricInterpretation =
903                        BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_Y_CB_CR;
904                } else if(numBands == 1 && sampleSize[0] == 8) {
905                    photometricInterpretation =
906                        BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_BLACK_IS_ZERO;
907                } else {
908                    throw new IIOException
909                        ("JPEG compression supported for 1- and 3-band byte images only!");
910                }
911                compressor = new TIFFJPEGCompressor(param);
912            } else if (compression ==
913                       BaselineTIFFTagSet.COMPRESSION_ZLIB) {
914                compressor = new TIFFZLibCompressor(param, predictor);
915            } else if (compression ==
916                       BaselineTIFFTagSet.COMPRESSION_PACKBITS) {
917                compressor = new TIFFPackBitsCompressor();
918            } else if (compression ==
919                       BaselineTIFFTagSet.COMPRESSION_DEFLATE) {
920                compressor = new TIFFDeflateCompressor(param, predictor);
921            } else {
922                // Determine inverse fill setting.
923                f = rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_FILL_ORDER);
924                boolean inverseFill = (f != null && f.getAsInt(0) == 2);
925
926                if(inverseFill) {
927                    compressor = new TIFFLSBCompressor();
928                } else {
929                    compressor = new TIFFNullCompressor();
930                }
931            } // compression == ?
932        } // compressor == null
933
934        if(DEBUG) {
935            if(param != null &&
936               param.getCompressionMode() == param.MODE_EXPLICIT) {
937                System.out.println("compressionType = "+
938                                   param.getCompressionType());
939            }
940            if(compressor != null) {
941                System.out.println("compressor = "+
942                                   compressor.getClass().getName());
943            }
944        }
945
946        if (colorConverter == null) {
947            if(cm != null &&
948               cm.getColorSpace().getType() == ColorSpace.TYPE_RGB) {
949                //
950                // Perform color conversion only if image has RGB color space.
951                //
952                if (photometricInterpretation ==
953                    BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_Y_CB_CR &&
954                    compression !=
955                    BaselineTIFFTagSet.COMPRESSION_JPEG) {
956                    //
957                    // Convert RGB to YCbCr only if compression type is not
958                    // JPEG in which case this is handled implicitly by the
959                    // compressor.
960                    //
961                    colorConverter =
962                        new TIFFYCbCrColorConverter(imageMetadata);
963                } else if (photometricInterpretation ==
964                           BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_CIELAB) {
965                    colorConverter = new TIFFCIELabColorConverter();
966                }
967            }
968        }
969
970        //
971        // Cannot at this time do YCbCr subsampling so set the
972        // YCbCrSubsampling field value to [1, 1] and the YCbCrPositioning
973        // field value to "cosited".
974        //
975        if(photometricInterpretation ==
976           BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_Y_CB_CR &&
977           compression !=
978           BaselineTIFFTagSet.COMPRESSION_JPEG) {
979            // Remove old subsampling and positioning fields.
980            rootIFD.removeTIFFField
981                (BaselineTIFFTagSet.TAG_Y_CB_CR_SUBSAMPLING);
982            rootIFD.removeTIFFField
983                (BaselineTIFFTagSet.TAG_Y_CB_CR_POSITIONING);
984
985            // Add unity chrominance subsampling factors.
986            rootIFD.addTIFFField
987                (new TIFFField
988                    (base.getTag(BaselineTIFFTagSet.TAG_Y_CB_CR_SUBSAMPLING),
989                     TIFFTag.TIFF_SHORT,
990                     2,
991                     new char[] {(char)1, (char)1}));
992
993            // Add cosited positioning.
994            rootIFD.addTIFFField
995                (new TIFFField
996                    (base.getTag(BaselineTIFFTagSet.TAG_Y_CB_CR_POSITIONING),
997                     TIFFTag.TIFF_SHORT,
998                     1,
999                     new char[] {
1000                         (char)BaselineTIFFTagSet.Y_CB_CR_POSITIONING_COSITED
1001                     }));
1002        }
1003
1004        TIFFField photometricInterpretationField =
1005            new TIFFField(
1006                base.getTag(BaselineTIFFTagSet.TAG_PHOTOMETRIC_INTERPRETATION),
1007                          photometricInterpretation);
1008        rootIFD.addTIFFField(photometricInterpretationField);
1009
1010        this.bitsPerSample = new char[numBands + numExtraSamples];
1011        this.bitDepth = 0;
1012        for (int i = 0; i < numBands; i++) {
1013            this.bitDepth = Math.max(bitDepth, sampleSize[i]);
1014        }
1015        if (bitDepth == 3) {
1016            bitDepth = 4;
1017        } else if (bitDepth > 4 && bitDepth < 8) {
1018            bitDepth = 8;
1019        } else if (bitDepth > 8 && bitDepth < 16) {
1020            bitDepth = 16;
1021        } else if (bitDepth > 16) {
1022            bitDepth = 32;
1023        }
1024
1025        for (int i = 0; i < bitsPerSample.length; i++) {
1026            bitsPerSample[i] = (char)bitDepth;
1027        }
1028
1029        // Emit BitsPerSample. If the image is bilevel, emit if and only
1030        // if already in the metadata and correct (count and value == 1).
1031        if (bitsPerSample.length != 1 || bitsPerSample[0] != 1) {
1032            TIFFField bitsPerSampleField =
1033                new TIFFField(
1034                           base.getTag(BaselineTIFFTagSet.TAG_BITS_PER_SAMPLE),
1035                           TIFFTag.TIFF_SHORT,
1036                           bitsPerSample.length,
1037                           bitsPerSample);
1038            rootIFD.addTIFFField(bitsPerSampleField);
1039        } else { // bitsPerSample.length == 1 && bitsPerSample[0] == 1
1040            TIFFField bitsPerSampleField =
1041                rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_BITS_PER_SAMPLE);
1042            if(bitsPerSampleField != null) {
1043                int[] bps = bitsPerSampleField.getAsInts();
1044                if(bps == null || bps.length != 1 || bps[0] != 1) {
1045                    rootIFD.removeTIFFField(BaselineTIFFTagSet.TAG_BITS_PER_SAMPLE);
1046                }
1047            }
1048        }
1049
1050        // Prepare SampleFormat field.
1051        f = rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_SAMPLE_FORMAT);
1052        if(f == null && (bitDepth == 16 || bitDepth == 32)) {
1053            // Set up default content for 16- and 32-bit cases.
1054            char sampleFormatValue;
1055            int dataType = sm.getDataType();
1056            if(bitDepth == 16 && dataType == DataBuffer.TYPE_USHORT) {
1057               sampleFormatValue =
1058                   (char)BaselineTIFFTagSet.SAMPLE_FORMAT_UNSIGNED_INTEGER;
1059            } else if(bitDepth == 32 && dataType == DataBuffer.TYPE_FLOAT) {
1060                sampleFormatValue =
1061                    (char)BaselineTIFFTagSet.SAMPLE_FORMAT_FLOATING_POINT;
1062            } else {
1063                sampleFormatValue =
1064                    BaselineTIFFTagSet.SAMPLE_FORMAT_SIGNED_INTEGER;
1065            }
1066            this.sampleFormat = (int)sampleFormatValue;
1067            char[] sampleFormatArray = new char[bitsPerSample.length];
1068            Arrays.fill(sampleFormatArray, sampleFormatValue);
1069
1070            // Update the metadata.
1071            TIFFTag sampleFormatTag =
1072                base.getTag(BaselineTIFFTagSet.TAG_SAMPLE_FORMAT);
1073
1074            TIFFField sampleFormatField =
1075                new TIFFField(sampleFormatTag, TIFFTag.TIFF_SHORT,
1076                              sampleFormatArray.length, sampleFormatArray);
1077
1078            rootIFD.addTIFFField(sampleFormatField);
1079        } else if(f != null) {
1080            // Get whatever was provided.
1081            sampleFormat = f.getAsInt(0);
1082        } else {
1083            // Set default value for internal use only.
1084            sampleFormat = BaselineTIFFTagSet.SAMPLE_FORMAT_UNDEFINED;
1085        }
1086
1087        if (extraSamples != null) {
1088            TIFFField extraSamplesField =
1089                new TIFFField(
1090                           base.getTag(BaselineTIFFTagSet.TAG_EXTRA_SAMPLES),
1091                           TIFFTag.TIFF_SHORT,
1092                           extraSamples.length,
1093                           extraSamples);
1094            rootIFD.addTIFFField(extraSamplesField);
1095        } else {
1096            rootIFD.removeTIFFField(BaselineTIFFTagSet.TAG_EXTRA_SAMPLES);
1097        }
1098
1099        TIFFField samplesPerPixelField =
1100            new TIFFField(
1101                         base.getTag(BaselineTIFFTagSet.TAG_SAMPLES_PER_PIXEL),
1102                         bitsPerSample.length);
1103        rootIFD.addTIFFField(samplesPerPixelField);
1104
1105        // Emit ColorMap if image is of palette color type
1106        if (photometricInterpretation ==
1107            BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_PALETTE_COLOR &&
1108            cm instanceof IndexColorModel) {
1109            char[] colorMap = new char[3*(1 << bitsPerSample[0])];
1110
1111            IndexColorModel icm = (IndexColorModel)cm;
1112
1113            // mapSize is determined by BitsPerSample, not by incoming ICM.
1114            int mapSize = 1 << bitsPerSample[0];
1115            int indexBound = Math.min(mapSize, icm.getMapSize());
1116            for (int i = 0; i < indexBound; i++) {
1117                colorMap[i] = (char)((icm.getRed(i)*65535)/255);
1118                colorMap[mapSize + i] = (char)((icm.getGreen(i)*65535)/255);
1119                colorMap[2*mapSize + i] = (char)((icm.getBlue(i)*65535)/255);
1120            }
1121
1122            TIFFField colorMapField =
1123                new TIFFField(
1124                           base.getTag(BaselineTIFFTagSet.TAG_COLOR_MAP),
1125                           TIFFTag.TIFF_SHORT,
1126                           colorMap.length,
1127                           colorMap);
1128            rootIFD.addTIFFField(colorMapField);
1129        } else {
1130            rootIFD.removeTIFFField(BaselineTIFFTagSet.TAG_COLOR_MAP);
1131        }
1132
1133        // Emit ICCProfile if there is no ICCProfile field already in the
1134        // metadata and the ColorSpace is non-standard ICC.
1135        if(cm != null &&
1136           rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_ICC_PROFILE) == null &&
1137           ImageUtil.isNonStandardICCColorSpace(cm.getColorSpace())) {
1138            ICC_ColorSpace iccColorSpace = (ICC_ColorSpace)cm.getColorSpace();
1139            byte[] iccProfileData = iccColorSpace.getProfile().getData();
1140            TIFFField iccProfileField =
1141                new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_ICC_PROFILE),
1142                              TIFFTag.TIFF_UNDEFINED,
1143                              iccProfileData.length,
1144                              iccProfileData);
1145            rootIFD.addTIFFField(iccProfileField);
1146        }
1147
1148        // Always emit XResolution and YResolution.
1149
1150        TIFFField XResolutionField =
1151            rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_X_RESOLUTION);
1152        TIFFField YResolutionField =
1153            rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_Y_RESOLUTION);
1154
1155        if(XResolutionField == null && YResolutionField == null) {
1156            long[][] resRational = new long[1][2];
1157            resRational[0] = new long[2];
1158
1159            TIFFField ResolutionUnitField =
1160                rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_RESOLUTION_UNIT);
1161
1162            // Don't force dimensionless if one of the other dimensional
1163            // quantities is present.
1164            if(ResolutionUnitField == null &&
1165               rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_X_POSITION) == null &&
1166               rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_Y_POSITION) == null) {
1167                // Set resolution to unit and units to dimensionless.
1168                resRational[0][0] = 1;
1169                resRational[0][1] = 1;
1170
1171                ResolutionUnitField =
1172                    new TIFFField(rootIFD.getTag
1173                                  (BaselineTIFFTagSet.TAG_RESOLUTION_UNIT),
1174                                  BaselineTIFFTagSet.RESOLUTION_UNIT_NONE);
1175                rootIFD.addTIFFField(ResolutionUnitField);
1176            } else {
1177                // Set resolution to a value which would make the maximum
1178                // image dimension equal to 4 inches as arbitrarily stated
1179                // in the description of ResolutionUnit in the TIFF 6.0
1180                // specification. If the ResolutionUnit field specifies
1181                // "none" then set the resolution to unity (1/1).
1182                int resolutionUnit = ResolutionUnitField != null ?
1183                    ResolutionUnitField.getAsInt(0) :
1184                    BaselineTIFFTagSet.RESOLUTION_UNIT_INCH;
1185                int maxDimension = Math.max(destWidth, destHeight);
1186                switch(resolutionUnit) {
1187                case BaselineTIFFTagSet.RESOLUTION_UNIT_INCH:
1188                    resRational[0][0] = maxDimension;
1189                    resRational[0][1] = 4;
1190                    break;
1191                case BaselineTIFFTagSet.RESOLUTION_UNIT_CENTIMETER:
1192                    resRational[0][0] = 100L*maxDimension; // divide out 100
1193                    resRational[0][1] = 4*254; // 2.54 cm/inch * 100
1194                    break;
1195                default:
1196                    resRational[0][0] = 1;
1197                    resRational[0][1] = 1;
1198                }
1199            }
1200
1201            XResolutionField =
1202                new TIFFField(rootIFD.getTag(BaselineTIFFTagSet.TAG_X_RESOLUTION),
1203                              TIFFTag.TIFF_RATIONAL,
1204                              1,
1205                              resRational);
1206            rootIFD.addTIFFField(XResolutionField);
1207
1208            YResolutionField =
1209                new TIFFField(rootIFD.getTag(BaselineTIFFTagSet.TAG_Y_RESOLUTION),
1210                              TIFFTag.TIFF_RATIONAL,
1211                              1,
1212                              resRational);
1213            rootIFD.addTIFFField(YResolutionField);
1214        } else if(XResolutionField == null && YResolutionField != null) {
1215            // Set XResolution to YResolution.
1216            long[] yResolution =
1217                (long[])YResolutionField.getAsRational(0).clone();
1218            XResolutionField =
1219             new TIFFField(rootIFD.getTag(BaselineTIFFTagSet.TAG_X_RESOLUTION),
1220                              TIFFTag.TIFF_RATIONAL,
1221                              1,
1222                              yResolution);
1223            rootIFD.addTIFFField(XResolutionField);
1224        } else if(XResolutionField != null && YResolutionField == null) {
1225            // Set YResolution to XResolution.
1226            long[] xResolution =
1227                (long[])XResolutionField.getAsRational(0).clone();
1228            YResolutionField =
1229             new TIFFField(rootIFD.getTag(BaselineTIFFTagSet.TAG_Y_RESOLUTION),
1230                              TIFFTag.TIFF_RATIONAL,
1231                              1,
1232                              xResolution);
1233            rootIFD.addTIFFField(YResolutionField);
1234        }
1235
1236        // Set mandatory fields, overriding metadata passed in
1237
1238        int width = destWidth;
1239        TIFFField imageWidthField =
1240            new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_IMAGE_WIDTH),
1241                          width);
1242        rootIFD.addTIFFField(imageWidthField);
1243
1244        int height = destHeight;
1245        TIFFField imageLengthField =
1246            new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_IMAGE_LENGTH),
1247                          height);
1248        rootIFD.addTIFFField(imageLengthField);
1249
1250        // Determine rowsPerStrip
1251
1252        int rowsPerStrip;
1253
1254        TIFFField rowsPerStripField =
1255            rootIFD.getTIFFField(BaselineTIFFTagSet.TAG_ROWS_PER_STRIP);
1256        if (rowsPerStripField != null) {
1257            rowsPerStrip = rowsPerStripField.getAsInt(0);
1258            if(rowsPerStrip < 0) {
1259                rowsPerStrip = height;
1260            }
1261        } else {
1262            int bitsPerPixel = bitDepth*(numBands + numExtraSamples);
1263            int bytesPerRow = (bitsPerPixel*width + 7)/8;
1264            rowsPerStrip =
1265                Math.max(Math.max(DEFAULT_BYTES_PER_STRIP/bytesPerRow, 1), 8);
1266        }

2760    private void locateIFD(int imageIndex, long[] ifdpos, long[] ifd) 
2761        throws IOException {
2762
2763        if(imageIndex < -1) {
2764            throw new IndexOutOfBoundsException("imageIndex < -1!");
2765        }

2768

2770        int byteOrder = stream.readUnsignedShort();

2773        } else if (byteOrder == 0x4949) {
2774            stream.setByteOrder(ByteOrder.LITTLE_ENDIAN);
2775        } else {

2780            stream.seek(startPos);
2781            throw new IIOException("Illegal magic number");
2782        }

2785        ifd[0] = stream.readUnsignedInt();

2792            }
2793            return;
2794        }
2795        stream.seek(ifd[0]);
2796
2797        for (int i = 0; imageIndex == -1 || i < imageIndex; i++) {
2798            int numFields;

2802                stream.seek(startPos);
2803                ifd[0] = 0;
