045package com.github.jaiimageio.impl.plugins.raw;
046
047import java.util.Locale;
048
049import javax.imageio.IIOException;
050import javax.imageio.ImageTypeSpecifier;
051import javax.imageio.ImageWriter;
052import javax.imageio.spi.ImageWriterSpi;
053import javax.imageio.spi.ServiceRegistry;
054
055import com.github.jaiimageio.impl.common.PackageUtil;
056
057public class RawImageWriterSpi extends ImageWriterSpi {
058    private static String [] readerSpiNames =
059        {"com.github.jaiimageio.impl.plugins.raw.RawImageReaderSpi"};
060    private static String[] formatNames = {"raw", "RAW"};
061    private static String[] entensions = {""};
062    private static String[] mimeType = {""};
063
064    private boolean registered = false;
065
066    public RawImageWriterSpi() {
067        super(PackageUtil.getVendor(),
068              PackageUtil.getVersion(),
069              formatNames,
070              entensions,
071              mimeType,
072              "com.github.jaiimageio.impl.plugins.raw.RawImageWriter",
073              STANDARD_OUTPUT_TYPE,
074              readerSpiNames,
075              true,
076              null, null, null, null,
077              true,
078              null, null, null, null);
079    }
080
081    public String getDescription(Locale locale) {
082        String desc = PackageUtil.getSpecificationTitle() + 
083            " Raw Image Writer";
084        return desc;
085    }
086
087    public void onRegistration(ServiceRegistry registry,
088                               Class category) {
089        if (registered) {
090            return;
091        }
092
093        registered = true;
094    }
095
096    public boolean canEncodeImage(ImageTypeSpecifier type) {
097        return true;
098    }
099
100    public ImageWriter createWriterInstance(Object extension)
101        throws IIOException {
102        return new RawImageWriter(this);
103    }
104}
