045package com.github.jaiimageio.impl.stream;
046
047import java.io.File;
048import java.io.IOException;
049import java.io.InputStream;
050import java.nio.channels.Channels;
051import java.nio.channels.FileChannel;
052import java.nio.channels.ReadableByteChannel;
053import java.util.Locale;
054
055import javax.imageio.spi.ImageInputStreamSpi;
056import javax.imageio.stream.FileCacheImageInputStream;
057import javax.imageio.stream.ImageInputStream;
058import javax.imageio.stream.MemoryCacheImageInputStream;
059
060import com.github.jaiimageio.impl.common.PackageUtil;
061import com.github.jaiimageio.stream.FileChannelImageInputStream;
062
063public class ChannelImageInputStreamSpi extends ImageInputStreamSpi {
064    public ChannelImageInputStreamSpi() {
065        super(PackageUtil.getVendor(),
066              PackageUtil.getVersion(),
067              ReadableByteChannel.class);
068    }
069
070    public ImageInputStream createInputStreamInstance(Object input,
071                                                      boolean useCache,
072                                                      File cacheDir)
073        throws IOException {
074
075        if(input == null ||
076           !(input instanceof ReadableByteChannel)) {
077            throw new IllegalArgumentException("XXX");
078        }
079
080        ImageInputStream stream = null;
081
082        if(input instanceof FileChannel) {
083            stream = new FileChannelImageInputStream((FileChannel)input);
084        } else {
085            InputStream inStream =
086                Channels.newInputStream((ReadableByteChannel)input);
087
088            if(useCache) {
089                try {
090                    stream = new FileCacheImageInputStream(inStream,
091                                                           cacheDir);
092                } catch(IOException e) {
093                    // Cache file could not be created.
094                }
095            }
096
097            if(stream == null) {
098                stream = new MemoryCacheImageInputStream(inStream);
099            }
100        }
101
102        return stream;
103    }
104
105    public String getDescription(Locale locale) {
106        return "NIO Channel ImageInputStream";
107    }
108}
