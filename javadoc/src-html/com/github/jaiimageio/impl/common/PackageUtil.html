062    /**

064     */
065    private static String vendor = "Sun Microsystems, Inc.";
066
067    /**
068     * Specification Title derived from Manifest.
069     */
070    private static String specTitle = "Java Advanced Imaging Image I/O Tools";
071
072    /**
073     * Set static flags.

075    static {
076        // Set codecLib flag.
077//        try {

081            // A Throwable is equivalent to unavailable. Throwable is used
082            // in case an Error rather than an Exception is thrown.
083            isCodecLibAvailable = false;
084//        }
085
086        // Set version and vendor strings.
087        try {
088            Class thisClass =
089                Class.forName("com.github.jaiimageio.impl.common.PackageUtil");
090            Package thisPackage = thisClass.getPackage();
091            if (thisPackage.getImplementationVersion()!= null && thisPackage.getImplementationVendor() != null) {
092                version = thisPackage.getImplementationVersion();
093                vendor = thisPackage.getImplementationVendor();
094                specTitle = thisPackage.getSpecificationTitle();
095            }
096        } catch(ClassNotFoundException e) {
097        }
098    }
099
100    /**
101     * Returns a <code>boolean</code> indicating whether codecLib is available.
102     */
103    public static final boolean isCodecLibAvailable() {
104        // Retrieve value of system property here to allow this to be
105        // modified dynamically.
106        Boolean result = (Boolean)
107            AccessController.doPrivileged(new PrivilegedAction() {
108                    public Object run() {
109                        String property = null;
110                        try {
111                            property =
112                                System.getProperty("com.github.jaiimageio.disableCodecLib");
113                        } catch(SecurityException se) {
114                            // Do nothing: leave 'property' null.
115                        }
116                        return (property != null &&
117                                property.equalsIgnoreCase("true")) ?
118                            Boolean.TRUE : Boolean.FALSE;
119                    }
120                });
121        boolean isCodecLibDisabled = result.booleanValue();
122
123        return isCodecLibAvailable && !isCodecLibDisabled;
124    }
125
126    /**
127     * Return a version string for the package.
128     */
129    public static final String getVersion() {
130        return version;
131    }
132
133    /**
134     * Return a vendor string for the package.
135     */
136    public static final String getVendor() {
137        return vendor;
138    }
139
140    /**
141     * Return the Specification Title string for the package.
142     */
143    public static final String getSpecificationTitle() {
144        return specTitle;
145    }
146}
