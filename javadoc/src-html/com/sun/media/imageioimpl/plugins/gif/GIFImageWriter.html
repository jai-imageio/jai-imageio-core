116    /**
117     * The number of bits represented by the value which should be a
118     * legal length for a color table.
119     */
120    private static int getNumBits(int value) throws IOException {
121        int numBits;
122        switch(value) {
123        case 2:
124            numBits = 1;
125            break;
126        case 4:
127            numBits = 2;
128            break;
129        case 8:
130            numBits = 3;
131            break;
132        case 16:
133            numBits = 4;
134            break;
135        case 32:
136            numBits = 5;
137            break;
138        case 64:
139            numBits = 6;
140            break;
141        case 128:
142            numBits = 7;
143            break;
144        case 256:
145            numBits = 8;
146            break;
147        default:
148            throw new IOException("Bad palette length: "+value+"!");
149        }
150        
151        return numBits;
152    }

154    /**
155     * Compute the source region and destination dimensions taking any
156     * parameter settings into account.
157     */
158    private static void computeRegions(Rectangle sourceBounds,
159                                       Dimension destSize,
160                                       ImageWriteParam p) {
161        ImageWriteParam param;
162        int periodX = 1;
163        int periodY = 1;
164        if (p != null) {
165            int[] sourceBands = p.getSourceBands();
166            if (sourceBands != null &&
167                (sourceBands.length != 1 ||
168                 sourceBands[0] != 0)) {
169                throw new IllegalArgumentException("Cannot sub-band image!");
170            }
171            
172            // Get source region and subsampling factors
173            Rectangle sourceRegion = p.getSourceRegion();
174            if (sourceRegion != null) {
175                // Clip to actual image bounds
176                sourceRegion = sourceRegion.intersection(sourceBounds);
177                sourceBounds.setBounds(sourceRegion);
178            }
179            
180            // Adjust for subsampling offsets
181            int gridX = p.getSubsamplingXOffset();
182            int gridY = p.getSubsamplingYOffset();
183            sourceBounds.x += gridX;
184            sourceBounds.y += gridY;
185            sourceBounds.width -= gridX;
186            sourceBounds.height -= gridY;
187            
188            // Get subsampling factors
189            periodX = p.getSourceXSubsampling();
190            periodY = p.getSourceYSubsampling();
191        }
192        
193        // Compute output dimensions
194        destSize.setSize((sourceBounds.width + periodX - 1)/periodX,
195                         (sourceBounds.height + periodY - 1)/periodY);
196        if (destSize.width <= 0 || destSize.height <= 0) {
197            throw new IllegalArgumentException("Empty source region!");
198        }
199    }

201    /**
202     * Create a color table from the image ColorModel and SampleModel.
203     */
204    private static byte[] createColorTable(ColorModel colorModel,
205                                           SampleModel sampleModel)
206    {
207        byte[] colorTable;
208        if (colorModel instanceof IndexColorModel) {
209            IndexColorModel icm = (IndexColorModel)colorModel;
210            int mapSize = icm.getMapSize();
211
212            /**
213             * The GIF image format assumes that size of image palette
214             * is power of two. We will use closest larger power of two 
215             * as size of color table.
216             */
217            int ctSize = getGifPaletteSize(mapSize);
218            
219            byte[] reds = new byte[ctSize];
220            byte[] greens = new byte[ctSize];
221            byte[] blues = new byte[ctSize];
222            icm.getReds(reds);
223            icm.getGreens(greens);
224            icm.getBlues(blues);
225
226            /**
227             * fill tail of color component arrays by replica of first color
228             * in order to avoid appearance of extra colors in the color table
229             */
230            for (int i = mapSize; i < ctSize; i++) {
231                reds[i] = reds[0];
232                greens[i] = greens[0];
233                blues[i] = blues[0];
234            }
235
236            colorTable = new byte[3*ctSize];
237            int idx = 0;
238            for (int i = 0; i < ctSize; i++) {
239                colorTable[idx++] = reds[i];
240                colorTable[idx++] = greens[i];
241                colorTable[idx++] = blues[i];
242            }
243        } else if (sampleModel.getNumBands() == 1) {
244            // create gray-scaled color table for single-banded images
245            int numBits = sampleModel.getSampleSize()[0];
246            if (numBits > 8) {
247                numBits = 8;
248            }
249            int colorTableLength = 3*(1 << numBits);
250            colorTable = new byte[colorTableLength];
251            for (int i = 0; i < colorTableLength; i++) {
252                colorTable[i] = (byte)(i/3);
253            }
254        } else {
255            // We do not have enough information here 
256            // to create well-fit color table for RGB image.
257            colorTable = null;
258        }
259        
260        return colorTable;
261    }
262    
263    /**
264     * According do GIF specification size of clor table (palette here) 
265     * must be in range from 2 to 256 and must be power of 2.
266     */
267    private static int getGifPaletteSize(int x) {
268        if (x <= 2) {
269            return 2;
270        }
271        x = x - 1;
272        x = x | (x >> 1);
273        x = x | (x >> 2);
274        x = x | (x >> 4);
275        x = x | (x >> 8);
276        x = x | (x >> 16);
277        return x + 1;
278    }

293    /**
294     * Merges <code>inData</code> into <code>outData</code>. The supplied
295     * metadata format name is attempted first and failing that the standard
296     * metadata format name is attempted.
297     */
298    private void convertMetadata(String metadataFormatName,
299                                 IIOMetadata inData,
300                                 IIOMetadata outData) {
301        String formatName = null;
302        
303        String nativeFormatName = inData.getNativeMetadataFormatName();
304        if (nativeFormatName != null &&
305            nativeFormatName.equals(metadataFormatName)) {
306            formatName = metadataFormatName;
307        } else {
308            String[] extraFormatNames = inData.getExtraMetadataFormatNames();
309            
310            if (extraFormatNames != null) {
311                for (int i = 0; i < extraFormatNames.length; i++) {
312                    if (extraFormatNames[i].equals(metadataFormatName)) {
313                        formatName = metadataFormatName;
314                        break;
315                    }
316                }
317            }
318        }
319        
320        if (formatName == null &&
321            inData.isStandardMetadataFormatSupported()) {
322            formatName = STANDARD_METADATA_NAME;
323        }
324        
325        if (formatName != null) {
326            try {
327                Node root = inData.getAsTree(formatName);
328                outData.mergeTree(formatName, root);
329            } catch(IIOInvalidTreeException e) {
330                // ignore
331            }
332        }
333    }

335    /**
336     * Creates a default stream metadata object and merges in the
337     * supplied metadata.
338     */
339    public IIOMetadata convertStreamMetadata(IIOMetadata inData,
340                                             ImageWriteParam param) {
341        if (inData == null) {
342            throw new IllegalArgumentException("inData == null!");
343        }
344        
345        IIOMetadata sm = getDefaultStreamMetadata(param);
346        
347        convertMetadata(STREAM_METADATA_NAME, inData, sm);
348        
349        return sm;
350    }

352    /**
353     * Creates a default image metadata object and merges in the
354     * supplied metadata.
355     */
356    public IIOMetadata convertImageMetadata(IIOMetadata inData,
357                                            ImageTypeSpecifier imageType,
358                                            ImageWriteParam param) {
359        if (inData == null) {
360            throw new IllegalArgumentException("inData == null!");
361        }
362        if (imageType == null) {
363            throw new IllegalArgumentException("imageType == null!");
364        }
365        
366        GIFWritableImageMetadata im =
367            (GIFWritableImageMetadata)getDefaultImageMetadata(imageType,
368                                                              param);
369        
370        // Save interlace flag state.
371        
372        boolean isProgressive = im.interlaceFlag;
373        
374        convertMetadata(IMAGE_METADATA_NAME, inData, im);
375        
376        // Undo change to interlace flag if not MODE_COPY_FROM_METADATA.
377        
378        if (param != null && param.canWriteProgressive() &&
379            param.getProgressiveMode() != param.MODE_COPY_FROM_METADATA) {
380            im.interlaceFlag = isProgressive;
381        }
382        
383        return im;
384    }

386    public void endWriteSequence() throws IOException {
387        if (stream == null) {
388            throw new IllegalStateException("output == null!");
389        }
390        if (!isWritingSequence) {
391            throw new IllegalStateException("prepareWriteSequence() was not invoked!");
392        }
393        writeTrailer();
394        resetLocal();      
395    }

397    public IIOMetadata getDefaultImageMetadata(ImageTypeSpecifier imageType,
398                                               ImageWriteParam param) {
399        GIFWritableImageMetadata imageMetadata =
400            new GIFWritableImageMetadata();
401        
402        // Image dimensions
403        
404        SampleModel sampleModel = imageType.getSampleModel();
405        
406        Rectangle sourceBounds = new Rectangle(sampleModel.getWidth(),
407                                               sampleModel.getHeight());
408        Dimension destSize = new Dimension();
409        computeRegions(sourceBounds, destSize, param);
410        
411        imageMetadata.imageWidth = destSize.width;
412        imageMetadata.imageHeight = destSize.height;
413        
414        // Interlacing
415
416        if (param != null && param.canWriteProgressive() &&
417            param.getProgressiveMode() == ImageWriteParam.MODE_DISABLED) {
418            imageMetadata.interlaceFlag = false;
419        } else {
420            imageMetadata.interlaceFlag = true;
421        }
422        
423        // Local color table
424
425        ColorModel colorModel = imageType.getColorModel();
426        
427        imageMetadata.localColorTable =
428            createColorTable(colorModel, sampleModel);
429        
430        // Transparency
431        
432        if (colorModel instanceof IndexColorModel) {
433            int transparentIndex =
434                ((IndexColorModel)colorModel).getTransparentPixel();
435            if (transparentIndex != -1) {
436                imageMetadata.transparentColorFlag = true;
437                imageMetadata.transparentColorIndex = transparentIndex;
438            }
439        }
440        
441        return imageMetadata;
442    }

455    public void prepareWriteSequence(IIOMetadata streamMetadata)
456      throws IOException {
457
458        if (stream == null) {
459            throw new IllegalStateException("Output is not set.");
460        }
461        
462        resetLocal();
463
464        // Save the possibly converted stream metadata as an instance variable.
465        if (streamMetadata == null) {
466            this.theStreamMetadata =
467                (GIFWritableStreamMetadata)getDefaultStreamMetadata(null);
468        } else {
469            this.theStreamMetadata = new GIFWritableStreamMetadata();
470            convertMetadata(STREAM_METADATA_NAME, streamMetadata,
471                            theStreamMetadata);
472        }
473
474        this.isWritingSequence = true;
475    }

506    public void write(IIOMetadata sm,
507                      IIOImage iioimage,
508                      ImageWriteParam p) throws IOException {
509        if (stream == null) {
510            throw new IllegalStateException("output == null!");
511        }
512        if (iioimage == null) {
513            throw new IllegalArgumentException("iioimage == null!");
514        }
515        if (iioimage.hasRaster()) {
516            throw new UnsupportedOperationException("canWriteRasters() == false!");
517        }
518        
519        resetLocal();
520        
521        GIFWritableStreamMetadata streamMetadata;
522        if (sm == null) {
523            streamMetadata =
524                (GIFWritableStreamMetadata)getDefaultStreamMetadata(p);
525        } else {
526            streamMetadata =
527                (GIFWritableStreamMetadata)convertStreamMetadata(sm, p);
528        }
529        
530        write(true, true, streamMetadata, iioimage, p);
531    }

533    public void writeToSequence(IIOImage image, ImageWriteParam param)
534      throws IOException {
535        if (stream == null) {
536            throw new IllegalStateException("output == null!");
537        }
538        if (image == null) {
539            throw new IllegalArgumentException("image == null!");
540        }
541        if (image.hasRaster()) {
542            throw new UnsupportedOperationException("canWriteRasters() == false!");
543        }
544        if (!isWritingSequence) {
545            throw new IllegalStateException("prepareWriteSequence() was not invoked!");
546        }
547        
548        write(!wroteSequenceHeader, false, theStreamMetadata,
549              image, param);
550        
551        if (!wroteSequenceHeader) {
552            wroteSequenceHeader = true;
553        }
554        
555        this.imageIndex++;
556    }

559    private boolean needToCreateIndex(RenderedImage image) {
560        
561        SampleModel sampleModel = image.getSampleModel();
562        ColorModel colorModel = image.getColorModel();
563        
564        return sampleModel.getNumBands() != 1 ||
565            sampleModel.getSampleSize()[0] > 8 ||
566            colorModel.getComponentSize()[0] > 8;
567    }

569    /**
570     * Writes any extension blocks, the Image Descriptor, the image data,
571     * and optionally the header (Signature and Logical Screen Descriptor)
572     * and trailer (Block Terminator).
573     *
574     * @param writeHeader Whether to write the header.
575     * @param writeTrailer Whether to write the trailer.
576     * @param sm The stream metadata or <code>null</code> if
577     * <code>writeHeader</code> is <code>false</code>.
578     * @param iioimage The image and image metadata.
579     * @param p The write parameters.
580     *
581     * @throws IllegalArgumentException if the number of bands is not 1.
582     * @throws IllegalArgumentException if the number of bits per sample is
583     * greater than 8.
584     * @throws IllegalArgumentException if the color component size is
585     * greater than 8.
586     * @throws IllegalArgumentException if <code>writeHeader</code> is
587     * <code>true</code> and <code>sm</code> is <code>null</code>.
588     * @throws IllegalArgumentException if <code>writeHeader</code> is
589     * <code>false</code> and a sequence is not being written.
590     */
591    private void write(boolean writeHeader,
592                       boolean writeTrailer,
593                       IIOMetadata sm,
594                       IIOImage iioimage,
595                       ImageWriteParam p) throws IOException {
596        clearAbortRequest();
597
598        RenderedImage image = iioimage.getRenderedImage();
599        
600        // Check for ability to encode image.
601        if (needToCreateIndex(image)) {
602            image = PaletteBuilder.createIndexedImage(image);
603            iioimage.setRenderedImage(image);
604        }
605
606        ColorModel colorModel = image.getColorModel();
607        SampleModel sampleModel = image.getSampleModel();
608        
609        // Determine source region and destination dimensions.
610        Rectangle sourceBounds = new Rectangle(image.getMinX(),
611                                               image.getMinY(),
612                                               image.getWidth(),
613                                               image.getHeight());
614        Dimension destSize = new Dimension();
615        computeRegions(sourceBounds, destSize, p);
616        
617        // Convert any provided image metadata.
618        GIFWritableImageMetadata imageMetadata = null;
619        if (iioimage.getMetadata() != null) {
620            imageMetadata = new GIFWritableImageMetadata();
621            convertMetadata(IMAGE_METADATA_NAME, iioimage.getMetadata(),
622                            imageMetadata);
623            // Converted rgb image can use palette different from global.
624            // In order to avoid color artefacts we want to be sure we use
625            // appropriate palette. For this we initialize local color table
626            // from current color and sample models.
627            // At this point we can guarantee that local color table can be
628            // build because image was already converted to indexed or
629            // gray-scale representations
630            if (imageMetadata.localColorTable == null) {
631                imageMetadata.localColorTable =
632                    createColorTable(colorModel, sampleModel);
633                
634                // in case of indexed image we should take care of 
635                // transparent pixels
636                if (colorModel instanceof IndexColorModel) {
637                    IndexColorModel icm =
638                        (IndexColorModel)colorModel;
639                    int index = icm.getTransparentPixel();
640                    imageMetadata.transparentColorFlag = (index != -1);
641                    if (imageMetadata.transparentColorFlag) {
642                        imageMetadata.transparentColorIndex = index;
643                    }
644                    /* NB: transparentColorFlag might have not beed reset for
645                       greyscale images but explicitly reseting it here
646                       is potentially not right thing to do until we have way
647                       to find whether current value was explicitly set by
648                       the user.
649                    */
650                }
651            }
652        }
653        
654        // Global color table values.
655        byte[] globalColorTable = null;
656        
657        // Write the header (Signature+Logical Screen Descriptor+
658        // Global Color Table).
659        if (writeHeader) {
660            if (sm == null) {
661                throw new IllegalArgumentException("Cannot write null header!");
662            }
663            
664            GIFWritableStreamMetadata streamMetadata =
665                (GIFWritableStreamMetadata)sm;
666            
667            // Set the version if not set.
668            if (streamMetadata.version == null) {
669                streamMetadata.version = "89a";
670            }
671            
672            // Set the Logical Screen Desriptor if not set.
673            if (streamMetadata.logicalScreenWidth == 
674                GIFMetadata.UNDEFINED_INTEGER_VALUE) 
675            {
676                streamMetadata.logicalScreenWidth = destSize.width;
677            }
678
679            if (streamMetadata.logicalScreenHeight == 
680                GIFMetadata.UNDEFINED_INTEGER_VALUE)
681            {
682                streamMetadata.logicalScreenHeight = destSize.height;
683            }
684
685            if (streamMetadata.colorResolution ==
686                GIFMetadata.UNDEFINED_INTEGER_VALUE)
687            {
688                streamMetadata.colorResolution = colorModel != null ?
689                    colorModel.getComponentSize()[0] :
690                    sampleModel.getSampleSize()[0];
691            }
692            
693            // Set the Global Color Table if not set, i.e., if not
694            // provided in the stream metadata.
695            if (streamMetadata.globalColorTable == null) {
696                if (isWritingSequence && imageMetadata != null &&
697                    imageMetadata.localColorTable != null) {
698                    // Writing a sequence and a local color table was
699                    // provided in the metadata of the first image: use it.
700                    streamMetadata.globalColorTable =
701                        imageMetadata.localColorTable;
702                } else if (imageMetadata == null ||
703                           imageMetadata.localColorTable == null) {
704                    // Create a color table.
705                    streamMetadata.globalColorTable =
706                        createColorTable(colorModel, sampleModel);
707                }
708            }
709
710            // Set the Global Color Table. At this point it should be
711            // A) the global color table provided in stream metadata, if any;
712            // B) the local color table of the image metadata, if any, if
713            //    writing a sequence;
714            // C) a table created on the basis of the first image ColorModel
715            //    and SampleModel if no local color table is available; or
716            // D) null if none of the foregoing conditions obtain (which
717            //    should only be if a sequence is not being written and
718            //    a local color table is provided in image metadata).
719            globalColorTable = streamMetadata.globalColorTable;
720            
721            // Write the header.
722            int bitsPerPixel;
723            if (globalColorTable != null) {
724                bitsPerPixel = getNumBits(globalColorTable.length/3);
725            } else if (imageMetadata != null &&
726                       imageMetadata.localColorTable != null) {
727                bitsPerPixel =
728                    getNumBits(imageMetadata.localColorTable.length/3);
729            } else {
730                bitsPerPixel = sampleModel.getSampleSize(0);
731            }
732            writeHeader(streamMetadata, bitsPerPixel);
733        } else if (isWritingSequence) {
734            globalColorTable = theStreamMetadata.globalColorTable;
735        } else {
736            throw new IllegalArgumentException("Must write header for single image!");
737        }
738        
739        // Write extension blocks, Image Descriptor, and image data.
740        writeImage(iioimage.getRenderedImage(), imageMetadata, p,
741                   globalColorTable, sourceBounds, destSize);
742        
743        // Write the trailer.
744        if (writeTrailer) {
745            writeTrailer();
746        }
747    }

749    /**
750     * Writes any extension blocks, the Image Descriptor, and the image data
751     *
752     * @param iioimage The image and image metadata.
753     * @param param The write parameters.
754     * @param globalColorTable The Global Color Table.
755     * @param sourceBounds The source region.
756     * @param destSize The destination dimensions.
757     */
758    private void writeImage(RenderedImage image,
759                            GIFWritableImageMetadata imageMetadata,
760                            ImageWriteParam param, byte[] globalColorTable,
761                            Rectangle sourceBounds, Dimension destSize)
762      throws IOException {
763        ColorModel colorModel = image.getColorModel();
764        SampleModel sampleModel = image.getSampleModel();
765        
766        boolean writeGraphicsControlExtension;
767        if (imageMetadata == null) {
768            // Create default metadata.
769            imageMetadata = (GIFWritableImageMetadata)getDefaultImageMetadata(
770                new ImageTypeSpecifier(image), param);
771            
772            // Set GraphicControlExtension flag only if there is
773            // transparency.
774            writeGraphicsControlExtension = imageMetadata.transparentColorFlag;
775        } else {
776            // Check for GraphicControlExtension element.
777            NodeList list = null;
778            try {
779                IIOMetadataNode root = (IIOMetadataNode)
780                    imageMetadata.getAsTree(IMAGE_METADATA_NAME);
781                list = root.getElementsByTagName("GraphicControlExtension");
782            } catch(IllegalArgumentException iae) {
783                // Should never happen.
784            }
785            
786            // Set GraphicControlExtension flag if element present.
787            writeGraphicsControlExtension =
788                list != null && list.getLength() > 0;
789            
790            // If progressive mode is not MODE_COPY_FROM_METADATA, ensure
791            // the interlacing is set per the ImageWriteParam mode setting.
792            if (param != null && param.canWriteProgressive()) {
793                if (param.getProgressiveMode() ==
794                    ImageWriteParam.MODE_DISABLED) {
795                    imageMetadata.interlaceFlag = false;
796                } else if (param.getProgressiveMode() ==
797                           ImageWriteParam.MODE_DEFAULT) {
798                    imageMetadata.interlaceFlag = true;
799                }
800            }
801        }
802        
803        // Unset local color table if equal to global color table.
804        if (Arrays.equals(globalColorTable, imageMetadata.localColorTable)) {
805            imageMetadata.localColorTable = null;
806        }
807      
808        // Override dimensions
809        imageMetadata.imageWidth = destSize.width;
810        imageMetadata.imageHeight = destSize.height;
811        
812        // Write Graphics Control Extension.
813        if (writeGraphicsControlExtension) {
814            writeGraphicControlExtension(imageMetadata);
815        }
816        
817        // Write extension blocks.
818        writePlainTextExtension(imageMetadata);
819        writeApplicationExtension(imageMetadata);
820        writeCommentExtension(imageMetadata);
821        
822        // Write Image Descriptor
823        int bitsPerPixel =
824            getNumBits(imageMetadata.localColorTable == null ?
825                       (globalColorTable == null ?
826                        sampleModel.getSampleSize(0) :
827                        globalColorTable.length/3) :
828                       imageMetadata.localColorTable.length/3);
829        writeImageDescriptor(imageMetadata, bitsPerPixel);
830        
831        // Write image data
832        writeRasterData(image, sourceBounds, destSize,
833                        param, imageMetadata.interlaceFlag);
834    }

836    private void writeRows(RenderedImage image, LZWCompressor compressor,
837                           int sx, int sdx, int sy, int sdy, int sw,
838                           int dy, int ddy, int dw, int dh,
839                           int numRowsWritten, int progressReportRowPeriod)
840      throws IOException {
841        if (DEBUG) System.out.println("Writing unoptimized");
842        
843        int[] sbuf = new int[sw];
844        byte[] dbuf = new byte[dw];
845        
846        Raster raster =
847            image.getNumXTiles() == 1 && image.getNumYTiles() == 1 ?
848            image.getTile(0, 0) : image.getData();
849        for (int y = dy; y < dh; y += ddy) {
850            if (numRowsWritten % progressReportRowPeriod == 0) {
851                if (abortRequested()) {
852                    processWriteAborted();
853                    return;
854                }
855                processImageProgress((numRowsWritten*100.0F)/dh);
856            }
857            
858            raster.getSamples(sx, sy, sw, 1, 0, sbuf);
859            for (int i = 0, j = 0; i < dw; i++, j += sdx) {
860                dbuf[i] = (byte)sbuf[j];
861            }
862            compressor.compress(dbuf, 0, dw);
863            numRowsWritten++;
864            sy += sdy;
865        }
866    }

868    private void writeRowsOpt(byte[] data, int offset, int lineStride,
869                              LZWCompressor compressor,
870                              int dy, int ddy, int dw, int dh,
871                              int numRowsWritten, int progressReportRowPeriod)
872      throws IOException {
873        if (DEBUG) System.out.println("Writing optimized");
874        
875        offset += dy*lineStride;
876        lineStride *= ddy;
877        for (int y = dy; y < dh; y += ddy) {
878            if (numRowsWritten % progressReportRowPeriod == 0) {
879                if (abortRequested()) {
880                    processWriteAborted();
881                    return;
882                }
883                processImageProgress((numRowsWritten*100.0F)/dh);
884            }
885            
886            compressor.compress(data, offset, dw);
887            numRowsWritten++;
888            offset += lineStride;
889        }
890    }

892    private void writeRasterData(RenderedImage image,
893                                 Rectangle sourceBounds,
894                                 Dimension destSize,
895                                 ImageWriteParam param,
896                                 boolean interlaceFlag) throws IOException {
897        
898        int sourceXOffset = sourceBounds.x;
899        int sourceYOffset = sourceBounds.y;
900        int sourceWidth = sourceBounds.width;
901        int sourceHeight = sourceBounds.height;
902        
903        int destWidth = destSize.width;
904        int destHeight = destSize.height;
905        
906        int periodX;
907        int periodY;
908        if (param == null) {
909            periodX = 1;
910            periodY = 1;
911        } else {
912            periodX = param.getSourceXSubsampling();
913            periodY = param.getSourceYSubsampling();
914        }
915        
916        SampleModel sampleModel = image.getSampleModel();
917        int bitsPerPixel = sampleModel.getSampleSize()[0];
918        
919        int initCodeSize = bitsPerPixel;
920        if (initCodeSize == 1) {
921            initCodeSize++;
922        }
923        stream.write(initCodeSize);
924        
925        LZWCompressor compressor =
926            new LZWCompressor(stream, initCodeSize, false);
927        
928        boolean isOptimizedCase =
929            periodX == 1 && periodY == 1 &&
930            sampleModel instanceof ComponentSampleModel &&
931            image.getNumXTiles() == 1 && image.getNumYTiles() == 1 &&
932            image.getTile(0, 0).getDataBuffer() instanceof DataBufferByte;
933        
934        int numRowsWritten = 0;
935        
936        int progressReportRowPeriod = Math.max(destHeight/20, 1);
937        
938        processImageStarted(imageIndex);
939        
940        if (interlaceFlag) {
941            if (DEBUG) System.out.println("Writing interlaced");
942            
943            if (isOptimizedCase) {
944                Raster tile = image.getTile(0, 0);
945                byte[] data = ((DataBufferByte)tile.getDataBuffer()).getData();
946                ComponentSampleModel csm =
947                    (ComponentSampleModel)tile.getSampleModel();
948                int offset = csm.getOffset(sourceXOffset -
949                                           tile.getSampleModelTranslateX(),
950                                           sourceYOffset -
951                                           tile.getSampleModelTranslateY(),
952                                           0);
953                int lineStride = csm.getScanlineStride();
954                
955                writeRowsOpt(data, offset, lineStride, compressor,
956                             0, 8, destWidth, destHeight,
957                             numRowsWritten, progressReportRowPeriod);
958                
959                if (abortRequested()) {
960                    return;
961                }
962                
963                numRowsWritten += destHeight/8;
964                
965                writeRowsOpt(data, offset, lineStride, compressor,
966                             4, 8, destWidth, destHeight,
967                             numRowsWritten, progressReportRowPeriod);
968                
969                if (abortRequested()) {
970                    return;
971                }
972                
973                numRowsWritten += (destHeight - 4)/8;
974                
975                writeRowsOpt(data, offset, lineStride, compressor,
976                             2, 4, destWidth, destHeight,
977                             numRowsWritten, progressReportRowPeriod);
978                
979                if (abortRequested()) {
980                    return;
981                }
982                
983                numRowsWritten += (destHeight - 2)/4;
984                
985                writeRowsOpt(data, offset, lineStride, compressor,
986                             1, 2, destWidth, destHeight,
987                             numRowsWritten, progressReportRowPeriod);
988            } else {
989                writeRows(image, compressor,
990                          sourceXOffset, periodX,
991                          sourceYOffset, 8*periodY,
992                          sourceWidth,
993                          0, 8, destWidth, destHeight,
994                          numRowsWritten, progressReportRowPeriod);
995                
996                if (abortRequested()) {
997                    return;
998                }
999                
1000                numRowsWritten += destHeight/8;
1001                
1002                writeRows(image, compressor, sourceXOffset, periodX,
1003                          sourceYOffset + 4*periodY, 8*periodY,
1004                          sourceWidth,
1005                          4, 8, destWidth, destHeight,
1006                          numRowsWritten, progressReportRowPeriod);
1007                
1008                if (abortRequested()) {
1009                    return;
1010                }
1011                
1012                numRowsWritten += (destHeight - 4)/8;
1013                
1014                writeRows(image, compressor, sourceXOffset, periodX,
1015                          sourceYOffset + 2*periodY, 4*periodY,
1016                          sourceWidth,
1017                          2, 4, destWidth, destHeight,
1018                          numRowsWritten, progressReportRowPeriod);
1019                
1020                if (abortRequested()) {
1021                    return;
1022                }
1023                
1024                numRowsWritten += (destHeight - 2)/4;
1025                
1026                writeRows(image, compressor, sourceXOffset, periodX,
1027                          sourceYOffset + periodY, 2*periodY,
1028                          sourceWidth,
1029                          1, 2, destWidth, destHeight,
1030                          numRowsWritten, progressReportRowPeriod);
1031            }
1032        } else {
1033            if (DEBUG) System.out.println("Writing non-interlaced");
1034            
1035            if (isOptimizedCase) {
1036                Raster tile = image.getTile(0, 0);
1037                byte[] data = ((DataBufferByte)tile.getDataBuffer()).getData();
1038                ComponentSampleModel csm =
1039                    (ComponentSampleModel)tile.getSampleModel();
1040                int offset = csm.getOffset(sourceXOffset -
1041                                           tile.getSampleModelTranslateX(),
1042                                           sourceYOffset -
1043                                           tile.getSampleModelTranslateY(),
1044                                           0);
1045                int lineStride = csm.getScanlineStride();
1046                
1047                writeRowsOpt(data, offset, lineStride, compressor,
1048                             0, 1, destWidth, destHeight,
1049                             numRowsWritten, progressReportRowPeriod);
1050            } else {
1051                writeRows(image, compressor,
1052                          sourceXOffset, periodX,
1053                          sourceYOffset, periodY,
1054                          sourceWidth,
1055                          0, 1, destWidth, destHeight,
1056                          numRowsWritten, progressReportRowPeriod);
1057            }
1058        }
1059        
1060        if (abortRequested()) {
1061            return;
1062        }
1063        
1064        processImageProgress(100.0F);
1065
1066        compressor.flush();
1067        
1068        stream.write(0x00);
1069        
1070        processImageComplete();
1071    }

1073    private void writeHeader(String version,
1074                             int logicalScreenWidth,
1075                             int logicalScreenHeight,
1076                             int colorResolution,
1077                             int pixelAspectRatio,
1078                             int backgroundColorIndex,
1079                             boolean sortFlag,
1080                             int bitsPerPixel,
1081                             byte[] globalColorTable) throws IOException {
1082        try {
1083            // Signature
1084            stream.writeBytes("GIF"+version);
1085            
1086            // Screen Descriptor
1087            // Width
1088            stream.writeShort((short)logicalScreenWidth);
1089            
1090            // Height
1091            stream.writeShort((short)logicalScreenHeight);
1092            
1093            // Global Color Table
1094            // Packed fields
1095            int packedFields = globalColorTable != null ? 0x80 : 0x00;
1096            packedFields |= ((colorResolution - 1) & 0x7) << 4;
1097            if (sortFlag) {
1098                packedFields |= 0x8;
1099            }
1100            packedFields |= (bitsPerPixel - 1);
1101            stream.write(packedFields);
1102            
1103            // Background color index
1104            stream.write(backgroundColorIndex);
1105
1106            // Pixel aspect ratio
1107            stream.write(pixelAspectRatio);
1108            
1109            // Global Color Table
1110            if (globalColorTable != null) {
1111                stream.write(globalColorTable);
1112            }
1113        } catch (IOException e) {
1114            throw new IIOException("I/O error writing header!", e);
1115        }
1116    }

1142    private void writeGraphicControlExtension(int disposalMethod,
1143                                              boolean userInputFlag,
1144                                              boolean transparentColorFlag,
1145                                              int delayTime,
1146                                              int transparentColorIndex)
1147      throws IOException {
1148        try {
1149            stream.write(0x21);
1150            stream.write(0xf9);
1151            
1152            stream.write(4);
1153            
1154            int packedFields = (disposalMethod & 0x3) << 2;
1155            if (userInputFlag) {
1156                packedFields |= 0x2;
1157            }
1158            if (transparentColorFlag) {
1159                packedFields |= 0x1;
1160            }
1161            stream.write(packedFields);
1162            
1163            stream.writeShort((short)delayTime);
1164            
1165            stream.write(transparentColorIndex);
1166            stream.write(0x00);
1167        } catch (IOException e) {
1168            throw new IIOException("I/O error writing Graphic Control Extension!", e);
1169        }
1170    }

1181    private void writeBlocks(byte[] data) throws IOException {
1182        if (data != null && data.length > 0) {
1183            int offset = 0;
1184            while (offset < data.length) {
1185                int len = Math.min(data.length - offset, 255);
1186                stream.write(len);
1187                stream.write(data, offset, len);
1188                offset += len;
1189            }
1190        }
1191    }

1220    private void writeApplicationExtension(GIFWritableImageMetadata im)
1221      throws IOException {
1222        if (im.applicationIDs != null) {
1223            Iterator iterIDs = im.applicationIDs.iterator();
1224            Iterator iterCodes = im.authenticationCodes.iterator();
1225            Iterator iterData = im.applicationData.iterator();
1226            
1227            while (iterIDs.hasNext()) {
1228                try {
1229                    stream.write(0x21);
1230                    stream.write(0xff);
1231                    
1232                    stream.write(11);
1233                    stream.write((byte[])iterIDs.next(), 0, 8);
1234                    stream.write((byte[])iterCodes.next(), 0, 3);
1235                    
1236                    writeBlocks((byte[])iterData.next());
1237                    
1238                    stream.write(0x00);
1239                } catch (IOException e) {
1240                    throw new IIOException("I/O error writing Application Extension!", e);
1241                }
1242            }
1243        }
