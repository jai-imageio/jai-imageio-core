711    public void decodeNextScanline(int lineIndex) throws IIOException {
712        int bits = 0, code = 0, isT = 0;
713        int current, entry, twoBits;
714        boolean isWhite = true;
715        int dstEnd = 0;
716
717        int bitOffset = 0;
718
719        // Initialize starting of the changing elements array
720        changingElemSize = 0;
721
722        // While scanline not complete
723        while (bitOffset < w) {
724
725            // Mark start of white run.
726            int runOffset = bitOffset;
727
728            while (isWhite && bitOffset < w) {
729                // White run
730                current = nextNBits(10);
731                entry = white[current];
732                
733                // Get the 3 fields from the entry
734                isT = entry & 0x0001;
735                bits = (entry >>> 1) & 0x0f;
736                
737                if (bits == 12) {          // Additional Make up code
738                    // Get the next 2 bits
739                    twoBits = nextLesserThan8Bits(2);
740                    // Consolidate the 2 new bits and last 2 bits into 4 bits
741                    current = ((current << 2) & 0x000c) | twoBits; 
742                    entry = additionalMakeup[current];
743                    bits = (entry >>> 1) & 0x07;     // 3 bits 0000 0111
744                    code  = (entry >>> 4) & 0x0fff;  // 12 bits
745                    bitOffset += code; // Skip white run
746
747                    updatePointer(4 - bits);
748                } else if (bits == 0) {     // ERROR
749                    warning("Error 0");
750                    // XXX return?
751                } else if (bits == 15) {    // EOL
752                    //
753                    // Instead of throwing an exception, assume that the
754                    // EOL was premature; emit a warning and return.
755                    //
756                    warning("Premature EOL in white run of line "+lineIndex+
757                            ": read "+bitOffset+" of "+w+" expected pixels.");
758                    return;
759                } else {
760                    // 11 bits - 0000 0111 1111 1111 = 0x07ff
761                    code = (entry >>> 5) & 0x07ff;  
762                    bitOffset += code;
763
764                    updatePointer(10 - bits);
765                    if (isT == 0) {
766                        isWhite = false;
767                        currChangingElems[changingElemSize++] = bitOffset;
768                    }
769                }
770            }
771
772            // Check whether this run completed one width
773            if (bitOffset == w) {
774                // If the white run has not been terminated then ensure that
775                // the next code word is a terminating code for a white run
776                // of length zero.
777                int runLength = bitOffset - runOffset;
778                if(isWhite &&
779                   runLength != 0 && runLength % 64 == 0 &&
780                   nextNBits(8) != 0x35) {
781                    warning("Missing zero white run length terminating code!");
782                    updatePointer(8);
783                }
784                break;
785            }
786
787            // Mark start of black run.
788            runOffset = bitOffset;
789
790            while (isWhite == false && bitOffset < w) {
791                // Black run
792                current = nextLesserThan8Bits(4);
793                entry = initBlack[current];
794                
795                // Get the 3 fields from the entry
796                isT = entry & 0x0001;
797                bits = (entry >>> 1) & 0x000f;
798                code = (entry >>> 5) & 0x07ff;
799
800                if (code == 100) {
801                    current = nextNBits(9);
802                    entry = black[current];
803                    
804                    // Get the 3 fields from the entry
805                    isT = entry & 0x0001;
806                    bits = (entry >>> 1) & 0x000f;
807                    code = (entry >>> 5) & 0x07ff;
808
809                    if (bits == 12) {
810                        // Additional makeup codes
811                        updatePointer(5);
812                        current = nextLesserThan8Bits(4);
813                        entry = additionalMakeup[current];
814                        bits = (entry >>> 1) & 0x07;     // 3 bits 0000 0111
815                        code  = (entry >>> 4) & 0x0fff;  // 12 bits
816
817                        setToBlack(bitOffset, code);
818                        bitOffset += code;
819
820                        updatePointer(4 - bits);
821                    } else if (bits == 15) {
822                        //
823                        // Instead of throwing an exception, assume that the
824                        // EOL was premature; emit a warning and return.
825                        //
826                        warning("Premature EOL in black run of line "+
827                                lineIndex+": read "+bitOffset+" of "+w+
828                                " expected pixels.");
829                        return;
830                    } else {
831                        setToBlack(bitOffset, code);
832                        bitOffset += code;
833
834                        updatePointer(9 - bits);
835                        if (isT == 0) {
836                            isWhite = true;
837                            currChangingElems[changingElemSize++] = bitOffset;
838                        }
839                    }
840                } else if (code == 200) {
841                    // Is a Terminating code
842                    current = nextLesserThan8Bits(2);
843                    entry = twoBitBlack[current];
844                    code = (entry >>> 5) & 0x07ff;
845                    bits = (entry >>> 1) & 0x0f;
846
847                    setToBlack(bitOffset, code);
848                    bitOffset += code;
849
850                    updatePointer(2 - bits);
851                    isWhite = true;
852                    currChangingElems[changingElemSize++] = bitOffset;
853                } else {
854                    // Is a Terminating code
855                    setToBlack(bitOffset, code);
856                    bitOffset += code;
857
858                    updatePointer(4 - bits);
859                    isWhite = true;
860                    currChangingElems[changingElemSize++] = bitOffset;
861                }
862            }
863
864            // Check whether this run completed one width
865            if (bitOffset == w) {
866                // If the black run has not been terminated then ensure that
867                // the next code word is a terminating code for a black run
868                // of length zero.
869                int runLength = bitOffset - runOffset;
870                if(!isWhite &&
871                   runLength != 0 && runLength % 64 == 0 &&
872                   nextNBits(10) != 0x37) {
873                    warning("Missing zero black run length terminating code!");
874                    updatePointer(10);
875                }
876                break;
877            }
878        }
879
880        currChangingElems[changingElemSize++] = bitOffset;
881    }

883    public void decodeT4() throws IIOException {
884        int height = h;
885
886        int a0, a1, b1, b2;
887        int[] b = new int[2];
888        int entry, code, bits, color;
889        boolean isWhite;
890        int currIndex = 0;
891        int temp[];
892
893        if(data.length < 2) {
894            throw new IIOException("Insufficient data to read initial EOL.");
895        }
896
897        // The data should start with an EOL code
898        int next12 = nextNBits(12);
899        if(next12 != 1) {
900            warning("T.4 compressed data should begin with EOL.");
901        }
902        updatePointer(12);
903
904        // Find the first one-dimensionally encoded line.
905        int modeFlag = 0;
906        int lines = -1; // indicates imaginary line before first actual line.
907        while(modeFlag != 1) {
908            try {
909                modeFlag = findNextLine();
910                lines++; // Normally 'lines' will be 0 on exiting loop.
911            } catch(EOFException eofe) {
912                throw new IIOException("No reference line present.");
913            }
914        }
915
916        int bitOffset;
917
918        // Then the 1D encoded scanline data will occur, changing elements
919        // array gets set.
920        decodeNextScanline(srcMinY);
921        lines++;
922        lineBitNum += bitsPerScanline;
923
924        while(lines < height) {
925
926            // Every line must begin with an EOL followed by a bit which
927            // indicates whether the following scanline is 1D or 2D encoded.
928            try {
929                modeFlag = findNextLine();
930            } catch(EOFException eofe) {
931                warning("Input exhausted before EOL found at line "+
932                        (srcMinY+lines)+": read 0 of "+w+" expected pixels.");
933                break;
934            }
935            if(modeFlag == 0) {
936                // 2D encoded scanline follows          
937                
938                // Initialize previous scanlines changing elements, and 
939                // initialize current scanline's changing elements array
940                temp = prevChangingElems;
941                prevChangingElems = currChangingElems;
942                currChangingElems = temp;
943                currIndex = 0;
944                
945                // a0 has to be set just before the start of this scanline.
946                a0 = -1;
947                isWhite = true;
948                bitOffset = 0;
949
950                lastChangingElement = 0;
951
952                while (bitOffset < w) {
953                    // Get the next changing element
954                    getNextChangingElement(a0, isWhite, b);
955                    
956                    b1 = b[0];
957                    b2 = b[1];
958                    
959                    // Get the next seven bits
960                    entry = nextLesserThan8Bits(7);
961                    
962                    // Run these through the 2DCodes table
963                    entry = (int)(twoDCodes[entry] & 0xff);
964                    
965                    // Get the code and the number of bits used up
966                    code = (entry & 0x78) >>> 3;
967                    bits = entry & 0x07;
968                    
969                    if (code == 0) {
970                        if (!isWhite) {
971                            setToBlack(bitOffset, b2 - bitOffset);
972                        }
973                        bitOffset = a0 = b2;
974                        
975                        // Set pointer to consume the correct number of bits.
976                        updatePointer(7 - bits);
977                    } else if (code == 1) {
978                        // Horizontal
979                        updatePointer(7 - bits);
980                        
981                        // identify the next 2 codes.
982                        int number;
983                        if (isWhite) {
984                            number = decodeWhiteCodeWord();
985                            bitOffset += number;
986                            currChangingElems[currIndex++] = bitOffset;
987                            
988                            number = decodeBlackCodeWord();
989                            setToBlack(bitOffset, number);
990                            bitOffset += number;
991                            currChangingElems[currIndex++] = bitOffset;
992                        } else {
993                            number = decodeBlackCodeWord();
994                            setToBlack(bitOffset, number);
995                            bitOffset += number;
996                            currChangingElems[currIndex++] = bitOffset;
997                            
998                            number = decodeWhiteCodeWord();
999                            bitOffset += number;
1000                            currChangingElems[currIndex++] = bitOffset;
1001                        }
1002                        
1003                        a0 = bitOffset;
1004                    } else if (code <= 8) {
1005                        // Vertical
1006                        a1 = b1 + (code - 5);
1007                        
1008                        currChangingElems[currIndex++] = a1;
1009                        
1010                        // We write the current color till a1 - 1 pos,
1011                        // since a1 is where the next color starts
1012                        if (!isWhite) {
1013                            setToBlack(bitOffset, a1 - bitOffset);
1014                        }
1015                        bitOffset = a0 = a1;
1016                        isWhite = !isWhite;
1017                            
1018                        updatePointer(7 - bits);
1019                    } else {
1020                        warning("Unknown coding mode encountered at line "+
1021                                (srcMinY+lines)+": read "+bitOffset+" of "+w+
1022                                " expected pixels.");
1023
1024                        // Find the next one-dimensionally encoded line.
1025                        int numLinesTested = 0;
1026                        while(modeFlag != 1) {
1027                            try {
1028                                modeFlag = findNextLine();
1029                                numLinesTested++;
1030                            } catch(EOFException eofe) {
1031                                warning("Sync loss at line "+
1032                                        (srcMinY+lines)+": read "+
1033                                        lines+" of "+height+" lines.");
1034                                return;
1035                            }
1036                        }
1037                        lines += numLinesTested - 1;
1038                        updatePointer(13);
1039                        break;
1040                    }
1041                }
1042                
1043                // Add the changing element beyond the current scanline for the
1044                // other color too
1045                currChangingElems[currIndex++] = bitOffset;
1046                changingElemSize = currIndex;
1047            } else { // modeFlag == 1
1048                // 1D encoded scanline follows
1049                decodeNextScanline(srcMinY+lines);
1050            }
1051
1052            lineBitNum += bitsPerScanline;
1053            lines++;
1054        } // while(lines < height)
1055    }

1095            // changing elements into the currChangingElems.
1096            temp = prevChangingElems;
1097            prevChangingElems = currChangingElems;
1098            cce = currChangingElems = temp;
1099            currIndex = 0;
1100
1101            // Start decoding the scanline
1102            bitOffset = 0;
1103
1104            // Reset search start position for getNextChangingElement
1105            lastChangingElement = 0;
1106
1107            // Till one whole scanline is decoded
1108            while (bitOffset < w) {
1109                // Get the next changing element 
1110                getNextChangingElement(a0, isWhite, b);
1111                b1 = b[0];
1112                b2 = b[1];
1113
1114                // Get the next seven bits
1115                entry = nextLesserThan8Bits(7);
1116                // Run these through the 2DCodes table
1117                entry = (int)(twoDCodes[entry] & 0xff);
1118
1119                // Get the code and the number of bits used up
1120                code = (entry & 0x78) >>> 3;
1121                bits = entry & 0x07;
1122                
1123                if (code == 0) { // Pass
1124                    // We always assume WhiteIsZero format for fax.
1125                    if (!isWhite) {
1126                        if(b2 > w) {
1127                            b2 = w;
1128                            warning("Decoded row "+(srcMinY+lines)+
1129                                    " too long; ignoring extra samples.");
1130                        }
1131                        setToBlack(bitOffset, b2 - bitOffset);
1132                    }
1133                    bitOffset = a0 = b2;
1134
1135                    // Set pointer to only consume the correct number of bits.
1136                    updatePointer(7 - bits);
1137                } else if (code == 1) { // Horizontal
1138                    // Set pointer to only consume the correct number of bits.
1139                    updatePointer(7 - bits);
1140                    
1141                    // identify the next 2 alternating color codes.
1142                    int number;
1143                    if (isWhite) {
1144                        // Following are white and black runs
1145                        number = decodeWhiteCodeWord();
1146                        bitOffset += number;
1147                        cce[currIndex++] = bitOffset;
1148
1149                        number = decodeBlackCodeWord();
1150                        if(number > w - bitOffset) {
1151                            number = w - bitOffset;
1152                            warning("Decoded row "+(srcMinY+lines)+
1153                                    " too long; ignoring extra samples.");
1154                        }
1155                        setToBlack(bitOffset, number);
1156                        bitOffset += number;
1157                        cce[currIndex++] = bitOffset;
1158                    } else {
1159                        // First a black run and then a white run follows
1160                        number = decodeBlackCodeWord();
1161                        if(number > w - bitOffset) {
1162                            number = w - bitOffset;
1163                            warning("Decoded row "+(srcMinY+lines)+
1164                                    " too long; ignoring extra samples.");
1165                        }
1166                        setToBlack(bitOffset, number);
1167                        bitOffset += number;
1168                        cce[currIndex++] = bitOffset;
1169
1170                        number = decodeWhiteCodeWord();
1171                        bitOffset += number;
1172                        cce[currIndex++] = bitOffset;
1173                    }
1174                    
1175                    a0 = bitOffset;
1176                } else if (code <= 8) { // Vertical
1177                    a1 = b1 + (code - 5);
1178                    cce[currIndex++] = a1;
1179
1180                    // We write the current color till a1 - 1 pos,
1181                    // since a1 is where the next color starts
1182                    if (!isWhite) {
1183                        if(a1 > w) {
1184                            a1 = w;
1185                            warning("Decoded row "+(srcMinY+lines)+
1186                                    " too long; ignoring extra samples.");
1187                        }
1188                        setToBlack(bitOffset, a1 - bitOffset);
1189                    }
1190                    bitOffset = a0 = a1;
1191                    isWhite = !isWhite;
1192
1193                    updatePointer(7 - bits);
1194                } else if (code == 11) {
1195                    int entranceCode = nextLesserThan8Bits(3);
1196                    if (entranceCode != 7) {
1197                        String msg =
1198                            "Unsupported entrance code "+entranceCode+
1199                            " for extension mode at line "+(srcMinY+lines)+".";
1200                        warning(msg);
1201                    }
1202
1203                    int zeros = 0;
1204                    boolean exit = false;
1205
1206                    while (!exit) {
1207                        while (nextLesserThan8Bits(1) != 1) {
1208                            zeros++;
1209                        }
1210                        
1211                        if (zeros > 5) {
1212                            // Exit code
1213                            
1214                            // Zeros before exit code
1215                            zeros = zeros - 6;
1216
1217                            if (!isWhite && (zeros > 0)) {
1218                                cce[currIndex++] = bitOffset;
1219                            }
1220                            
1221                            // Zeros before the exit code
1222                            bitOffset += zeros;
1223                            if (zeros > 0) {
1224                                // Some zeros have been written
1225                                isWhite = true;
1226                            } 
1227                            
1228                            // Read in the bit which specifies the color of
1229                            // the following run
1230                            if (nextLesserThan8Bits(1) == 0) {
1231                                if (!isWhite) {
1232                                    cce[currIndex++] = bitOffset;
1233                                }
1234                                isWhite = true;
1235                            } else {
1236                                if (isWhite) {
1237                                    cce[currIndex++] = bitOffset;
1238                                }
1239                                isWhite = false;
1240                            }
1241
1242                            exit = true;
1243                        } 
1244
1245                        if (zeros == 5) {
1246                            if (!isWhite) {
1247                                cce[currIndex++] = bitOffset;
1248                            }
1249                            bitOffset += zeros;
1250
1251                            // Last thing written was white
1252                            isWhite = true;
1253                        } else {
1254                            bitOffset += zeros;
1255
1256                            cce[currIndex++] = bitOffset;
1257                            setToBlack(bitOffset, 1);
1258                            ++bitOffset;
1259
1260                            // Last thing written was black
1261                            isWhite = false;
1262                        }
1263
1264                    }
1265                } else {
1266                    String msg =
1267                        "Unknown coding mode encountered at line "+
1268                        (srcMinY+lines)+".";
1269                    warning(msg);
1270                }
1271            } // while bitOffset < w
1272            
1273            // Add the changing element beyond the current scanline for the
1274            // other color too, if not already added previously
1275            if (currIndex <= w)
1276                cce[currIndex++] = bitOffset;
1277
1278            // Number of changing elements in this scanline.
1279            changingElemSize = currIndex;
1280
1281            lineBitNum += bitsPerScanline;
1282        } // for lines < height
1283    }

1285    private void setToBlack(int bitNum, int numBits) {
1286        // bitNum is relative to current scanline so bump it by lineBitNum
1287        bitNum += lineBitNum;
1288
1289        int lastBit = bitNum + numBits;
1290        int byteNum = bitNum >> 3;
1291
1292        // Handle bits in first byte
1293        int shift = bitNum & 0x7;
1294        if (shift > 0) {
1295            int maskVal = 1 << (7 - shift);
1296            byte val = buffer[byteNum];
1297            while (maskVal > 0 && bitNum < lastBit) {
1298                val |= maskVal;
1299                maskVal >>= 1;
1300                ++bitNum;
1301            }
1302            buffer[byteNum] = val;
1303        }
1304
1305        // Fill in 8 bits at a time
1306        byteNum = bitNum >> 3;
1307        while (bitNum < lastBit - 7) {
1308            buffer[byteNum++] = (byte)255;
1309            bitNum += 8;
1310        }
1311
1312        // Fill in remaining bits
1313        while (bitNum < lastBit) {
1314            byteNum = bitNum >> 3;
1315            buffer[byteNum] |= 1 << (7 - (bitNum & 0x7));
1316            ++bitNum;
1317        }
1318    }

1320    // Returns run length
1321    private int decodeWhiteCodeWord() throws IIOException {
1322        int current, entry, bits, isT, twoBits, code = -1;
1323        int runLength = 0;
1324        boolean isWhite = true;
1325
1326        while (isWhite) {
1327            current = nextNBits(10);
1328            entry = white[current];
1329            
1330            // Get the 3 fields from the entry
1331            isT = entry & 0x0001;
1332            bits = (entry >>> 1) & 0x0f;
1333            
1334            if (bits == 12) {           // Additional Make up code
1335                // Get the next 2 bits
1336                twoBits = nextLesserThan8Bits(2);
1337                // Consolidate the 2 new bits and last 2 bits into 4 bits
1338                current = ((current << 2) & 0x000c) | twoBits; 
1339                entry = additionalMakeup[current];
1340                bits = (entry >>> 1) & 0x07;     // 3 bits 0000 0111
1341                code = (entry >>> 4) & 0x0fff;   // 12 bits
1342                runLength += code;
1343                updatePointer(4 - bits);
1344            } else if (bits == 0) {     // ERROR
1345                throw new IIOException("Error 0");
1346            } else if (bits == 15) {    // EOL
1347                throw new IIOException("Error 1");
1348            } else {
1349                // 11 bits - 0000 0111 1111 1111 = 0x07ff
1350                code = (entry >>> 5) & 0x07ff;  
1351                runLength += code;
1352                updatePointer(10 - bits);
1353                if (isT == 0) {
1354                    isWhite = false;
1355                }
1356            }
1357        }
1358
1359        return runLength;
1360    }

1362    // Returns run length
1363    private int decodeBlackCodeWord() throws IIOException {
1364        int current, entry, bits, isT, twoBits, code = -1;
1365        int runLength = 0;
1366        boolean isWhite = false;
1367        
1368        while (!isWhite) {
1369            current = nextLesserThan8Bits(4);
1370            entry = initBlack[current];
1371            
1372            // Get the 3 fields from the entry
1373            isT = entry & 0x0001;
1374            bits = (entry >>> 1) & 0x000f;
1375            code = (entry >>> 5) & 0x07ff;
1376            
1377            if (code == 100) {
1378                current = nextNBits(9);
1379                entry = black[current];
1380                
1381                // Get the 3 fields from the entry
1382                isT = entry & 0x0001;
1383                bits = (entry >>> 1) & 0x000f;
1384                code = (entry >>> 5) & 0x07ff;
1385                
1386                if (bits == 12) {
1387                    // Additional makeup codes              
1388                    updatePointer(5);
1389                    current = nextLesserThan8Bits(4);
1390                    entry = additionalMakeup[current];
1391                    bits = (entry >>> 1) & 0x07;     // 3 bits 0000 0111
1392                    code  = (entry >>> 4) & 0x0fff;  // 12 bits
1393                    runLength += code;
1394                    
1395                    updatePointer(4 - bits);
1396                } else if (bits == 15) {
1397                    // EOL code
1398                    throw new IIOException("Error 2");
1399                } else {
1400                    runLength += code;
1401                    updatePointer(9 - bits);
1402                    if (isT == 0) {
1403                        isWhite = true;
1404                    }
1405                }
1406            } else if (code == 200) {
1407                // Is a Terminating code
1408                current = nextLesserThan8Bits(2);
1409                entry = twoBitBlack[current];
1410                code = (entry >>> 5) & 0x07ff;
1411                runLength += code;
1412                bits = (entry >>> 1) & 0x0f;
1413                updatePointer(2 - bits);
1414                isWhite = true;
1415            } else {
1416                // Is a Terminating code
1417                runLength += code;
1418                updatePointer(4 - bits);
1419                isWhite = true;
1420            }
1421        }
1422
1423        return runLength;
1424    }

1426    private int findNextLine() throws IIOException, EOFException {
1427        // Set maximum and current bit index into the compressed data.
1428        int bitIndexMax = data.length*8 - 1;
1429        int bitIndexMax12 = bitIndexMax - 12;
1430        int bitIndex = bytePointer*8 + bitPointer;
1431
1432        // Loop while at least 12 bits are available.
1433        while(bitIndex <= bitIndexMax12) {
1434            // Get the next 12 bits.
1435            int next12Bits = nextNBits(12);
1436            bitIndex += 12;
1437
1438            // Loop while the 12 bits are not unity, i.e., while the EOL
1439            // has not been reached, and there is at least one bit left.
1440            while(next12Bits != 1 && bitIndex < bitIndexMax) {
1441                next12Bits =
1442                    ((next12Bits & 0x000007ff) << 1) |
1443                    (nextLesserThan8Bits(1) & 0x00000001);
1444                bitIndex++;
1445            }
1446
1447            if(next12Bits == 1) { // now positioned just after EOL
1448                if(oneD == 1) { // two-dimensional coding
1449                    if(bitIndex < bitIndexMax) {
1450                        // check next bit against type of line being sought
1451                        return nextLesserThan8Bits(1);
1452                    }
1453                } else {
1454                    return 1;
1455                }
1456            }
1457        }
1458            
1459        // EOL not found.
1460        throw new EOFException();
1461    }

1463    private void getNextChangingElement(int a0, boolean isWhite, int[] ret) throws IIOException {
1464        // Local copies of instance variables
1465        int[] pce = this.prevChangingElems;
1466        int ces = this.changingElemSize;
1467
1468        // If the previous match was at an odd element, we still
1469        // have to search the preceeding element.
1470        // int start = lastChangingElement & ~0x1;
1471        int start = lastChangingElement > 0 ? lastChangingElement - 1 : 0;
1472        if (isWhite) {
1473            start &= ~0x1; // Search even numbered elements
1474        } else {
1475            start |= 0x1; // Search odd numbered elements
1476        }
1477
1478        int i = start;
1479        for (; i < ces; i += 2) {
1480            int temp = pce[i];
1481            if (temp > a0) {
1482                lastChangingElement = i;
1483                ret[0] = temp;
1484                break;
1485            }
1486        }
1487
1488        if (i + 1 < ces) {
1489            ret[1] = pce[i + 1];
1490        }
1491    }

1493    private int nextNBits(int bitsToGet) throws IIOException {
1494        byte b, next, next2next;
1495        int l = data.length - 1;
1496        int bp = this.bytePointer;
1497            
1498        if (fillOrder == 1) {
1499            b = data[bp];
1500            
1501            if (bp == l) {
1502                next = 0x00;
1503                next2next = 0x00;
1504            } else if ((bp + 1) == l) {
1505                next = data[bp + 1];
1506                next2next = 0x00;
1507            } else {
1508                next = data[bp + 1];
1509                next2next = data[bp + 2];
1510            }
1511        } else if (fillOrder == 2) {
1512            b = flipTable[data[bp] & 0xff];
1513
1514            if (bp == l) {
1515                next = 0x00;
1516                next2next = 0x00;
1517            } else if ((bp + 1) == l) {
1518                next = flipTable[data[bp + 1] & 0xff];
1519                next2next = 0x00;
1520            } else {
1521                next = flipTable[data[bp + 1] & 0xff];
1522                next2next = flipTable[data[bp + 2] & 0xff];
1523            }
1524        } else {
1525            throw new IIOException("Invalid FillOrder");
1526        }
1527
1528        int bitsLeft = 8 - bitPointer;
1529        int bitsFromNextByte = bitsToGet - bitsLeft;
1530        int bitsFromNext2NextByte = 0;
1531        if (bitsFromNextByte > 8) {
1532            bitsFromNext2NextByte = bitsFromNextByte - 8;
1533            bitsFromNextByte = 8;
1534        }
1535        
1536        bytePointer++;
1537
1538        int i1 = (b & table1[bitsLeft]) << (bitsToGet - bitsLeft);
1539        int i2 = (next & table2[bitsFromNextByte]) >>> (8 - bitsFromNextByte);
1540        
1541        int i3 = 0;
1542        if (bitsFromNext2NextByte != 0) {
1543            i2 <<= bitsFromNext2NextByte;
1544            i3 = (next2next & table2[bitsFromNext2NextByte]) >>> 
1545                (8 - bitsFromNext2NextByte);
1546            i2 |= i3;
1547            bytePointer++;
1548            bitPointer = bitsFromNext2NextByte;
1549        } else {
1550            if (bitsFromNextByte == 8) {
1551                bitPointer = 0;
1552                bytePointer++;
1553            } else {
1554                bitPointer = bitsFromNextByte;
1555            }
1556        }
1557        
1558        int i = i1 | i2;
1559        return i;
1560    }

1562    private int nextLesserThan8Bits(int bitsToGet) throws IIOException {
1563        byte b, next;
1564        int l = data.length - 1;
1565        int bp = this.bytePointer;
1566
1567        if (fillOrder == 1) {
1568            b = data[bp];
1569            if (bp == l) {
1570                next = 0x00;
1571            } else {
1572                next = data[bp + 1];
1573            }
1574        } else if (fillOrder == 2) {
1575            b = flipTable[data[bp] & 0xff];
1576            if (bp == l) {
1577                next = 0x00;
1578            } else {
1579                next = flipTable[data[bp + 1] & 0xff];
1580            }
1581        } else {
1582            throw new IIOException("Invalid FillOrder");
1583        }
1584
1585        int bitsLeft = 8 - bitPointer;
1586        int bitsFromNextByte = bitsToGet - bitsLeft;
1587
1588        int shift = bitsLeft - bitsToGet;
1589        int i1, i2;
1590        if (shift >= 0) {
1591            i1 = (b & table1[bitsLeft]) >>> shift;
1592            bitPointer += bitsToGet;
1593            if (bitPointer == 8) {
1594                bitPointer = 0;
1595                bytePointer++;
1596            }
1597        } else {
1598            i1 = (b & table1[bitsLeft]) << (-shift);
1599            i2 = (next & table2[bitsFromNextByte]) >>> (8 - bitsFromNextByte);
1600
1601            i1 |= i2;
1602            bytePointer++;
1603            bitPointer = bitsFromNextByte;
1604        }
1605
1606        return i1;
1607    }

1611        if (bitsToMoveBack > 8) {
1612            bytePointer -= bitsToMoveBack/8;

1614        }
1615

1617        if (i < 0) {
1618            bytePointer--;
1619            bitPointer = 8 + i;
1620        } else {
1621            bitPointer = i;
1622        }
