045package com.sun.media.imageioimpl.plugins.tiff;
046
047import java.util.Collection;
048import java.util.HashMap;
049import java.util.Iterator;
050import java.util.Locale;
051import java.util.Map;
052import java.util.MissingResourceException;
053import java.util.ResourceBundle;
054import javax.imageio.ImageTypeSpecifier;
055import javax.imageio.metadata.IIOMetadataFormat;
056import com.sun.media.imageio.plugins.tiff.BaselineTIFFTagSet;
057import com.sun.media.imageio.plugins.tiff.TIFFTag;
058import com.sun.media.imageio.plugins.tiff.TIFFTagSet;
059
060public abstract class TIFFMetadataFormat implements IIOMetadataFormat {
061
062    protected Map elementInfoMap = new HashMap();
063    protected Map attrInfoMap = new HashMap();
064
065    protected String resourceBaseName;
066    protected String rootName;
067
068    public String getRootName() {
069        return rootName;
070    }
071
072    private String getResource(String key, Locale locale) {
073        if (locale == null) {
074            locale = Locale.getDefault();
075        }
076        try {
077            ResourceBundle bundle =
078                ResourceBundle.getBundle(resourceBaseName, locale);
079            return bundle.getString(key);
080        } catch (MissingResourceException e) {
081            return null;
082        }
083    }
084
085    private TIFFElementInfo getElementInfo(String elementName) {
086        if (elementName == null) {
087            throw new IllegalArgumentException("elementName == null!");
088        }
089        TIFFElementInfo info =
090            (TIFFElementInfo)elementInfoMap.get(elementName);
091        if (info == null) {
092            throw new IllegalArgumentException("No such element: " +
093                                               elementName);
094        }
095        return info;
096    }
097
098    private TIFFAttrInfo getAttrInfo(String elementName, String attrName) {
099        if (elementName == null) {
100            throw new IllegalArgumentException("elementName == null!");
101        }
102        if (attrName == null) {
103            throw new IllegalArgumentException("attrName == null!");
104        }
105        String key = elementName + "/" + attrName;
106        TIFFAttrInfo info = (TIFFAttrInfo)attrInfoMap.get(key);
107        if (info == null) {
108            throw new IllegalArgumentException("No such attribute: " + key);
109        }
110        return info;
111    }
112
113    public int getElementMinChildren(String elementName) {
114        TIFFElementInfo info = getElementInfo(elementName);
115        return info.minChildren;
116    }
117
118    public int getElementMaxChildren(String elementName) {
119        TIFFElementInfo info = getElementInfo(elementName);
120        return info.maxChildren;
121    }
122
123    public String getElementDescription(String elementName, Locale locale) {
124        if (!elementInfoMap.containsKey(elementName)) {
125            throw new IllegalArgumentException("No such element: " +
126                                               elementName);
127        }
128        return getResource(elementName, locale);
129    }
130
131    public int getChildPolicy(String elementName) {
132        TIFFElementInfo info = getElementInfo(elementName);
133        return info.childPolicy;
134    }
135
136    public String[] getChildNames(String elementName) {
137        TIFFElementInfo info = getElementInfo(elementName);
138        return info.childNames;
139    }
140
141    public String[] getAttributeNames(String elementName) {
142        TIFFElementInfo info = getElementInfo(elementName);
143        return info.attributeNames;
144    }
145
146    public int getAttributeValueType(String elementName, String attrName) {
147        TIFFAttrInfo info = getAttrInfo(elementName, attrName);
148        return info.valueType;
149    }
150
151    public int getAttributeDataType(String elementName, String attrName) {
152        TIFFAttrInfo info = getAttrInfo(elementName, attrName);
153        return info.dataType;
154    }
155
156    public boolean isAttributeRequired(String elementName, String attrName) {
157        TIFFAttrInfo info = getAttrInfo(elementName, attrName);
158        return info.isRequired;
159    }
160
161    public String getAttributeDefaultValue(String elementName,
162                                           String attrName) {
163        TIFFAttrInfo info = getAttrInfo(elementName, attrName);
164        return info.defaultValue;
165    }
166
167    public String[] getAttributeEnumerations(String elementName,
168                                             String attrName) {
169        TIFFAttrInfo info = getAttrInfo(elementName, attrName);
170        return info.enumerations;
171    }
172
173    public String getAttributeMinValue(String elementName, String attrName) {
174        TIFFAttrInfo info = getAttrInfo(elementName, attrName);
175        return info.minValue;
176    }
177
178    public String getAttributeMaxValue(String elementName, String attrName) {
179        TIFFAttrInfo info = getAttrInfo(elementName, attrName);
180        return info.maxValue;
181    }
182
183    public int getAttributeListMinLength(String elementName, String attrName) {
184        TIFFAttrInfo info = getAttrInfo(elementName, attrName);
185        return info.listMinLength;
186    }
187
188    public int getAttributeListMaxLength(String elementName, String attrName) {
189        TIFFAttrInfo info = getAttrInfo(elementName, attrName);
190        return info.listMaxLength;
191    }
192
193    public String getAttributeDescription(String elementName, String attrName,
194                                          Locale locale) {
195        String key = elementName + "/" + attrName;
196        if (!attrInfoMap.containsKey(key)) {
197            throw new IllegalArgumentException("No such attribute: " + key);
198        }
199        return getResource(key, locale);
200    }
201
202    public int getObjectValueType(String elementName) {
203        TIFFElementInfo info = getElementInfo(elementName);
204        return info.objectValueType;
205    }
206
207    public Class getObjectClass(String elementName) {
208        TIFFElementInfo info = getElementInfo(elementName);
209        if (info.objectValueType == VALUE_NONE) {
210            throw new IllegalArgumentException(
211                     "Element cannot contain an object value: " + elementName);
212        }
213        return info.objectClass;
214    }
215
216    public Object getObjectDefaultValue(String elementName) {
217        TIFFElementInfo info = getElementInfo(elementName);
218        if (info.objectValueType == VALUE_NONE) {
219            throw new IllegalArgumentException(
220                     "Element cannot contain an object value: " + elementName);
221        }
222        return info.objectDefaultValue;
223    }
224
225    public Object[] getObjectEnumerations(String elementName) {
226        TIFFElementInfo info = getElementInfo(elementName);
227        if (info.objectValueType == VALUE_NONE) {
228            throw new IllegalArgumentException(
229                     "Element cannot contain an object value: " + elementName);
230        }
231        return info.objectEnumerations;
232    }
233
234    public Comparable getObjectMinValue(String elementName) {
235        TIFFElementInfo info = getElementInfo(elementName);
236        if (info.objectValueType == VALUE_NONE) {
237            throw new IllegalArgumentException(
238                     "Element cannot contain an object value: " + elementName);
239        }
240        return info.objectMinValue;
241    }
242
243    public Comparable getObjectMaxValue(String elementName) {
244        TIFFElementInfo info = getElementInfo(elementName);
245        if (info.objectValueType == VALUE_NONE) {
246            throw new IllegalArgumentException(
247                     "Element cannot contain an object value: " + elementName);
248        }
249        return info.objectMaxValue;
250    }
251
252    public int getObjectArrayMinLength(String elementName) {
253        TIFFElementInfo info = getElementInfo(elementName);
254        if (info.objectValueType == VALUE_NONE) {
255            throw new IllegalArgumentException(
256                     "Element cannot contain an object value: " + elementName);
257        }
258        return info.objectArrayMinLength;
259    }
260
261    public int getObjectArrayMaxLength(String elementName) {
262        TIFFElementInfo info = getElementInfo(elementName);
263        if (info.objectValueType == VALUE_NONE) {
264            throw new IllegalArgumentException(
265                     "Element cannot contain an object value: " + elementName);
266        }
267        return info.objectArrayMaxLength;
268    }
269
270    public TIFFMetadataFormat() {}
271}
