045package com.sun.media.imageio.plugins.tiff;
046
047import java.io.IOException;
048import javax.imageio.ImageWriter;
049import javax.imageio.metadata.IIOMetadata;
050import javax.imageio.stream.ImageOutputStream;
051import com.sun.media.imageioimpl.plugins.tiff.TIFFImageWriter;
052
053/**
054 * An abstract superclass for pluggable TIFF compressors.
055 */
056public abstract class TIFFCompressor {
057
058    /**
059     * The <code>ImageWriter</code> calling this
060     * <code>TIFFCompressor</code>.
061     */
062    protected ImageWriter writer;
063
064    /**
065     * The <code>IIOMetadata</code> object containing metadata for the
066     * current image.
067     */
068    protected IIOMetadata metadata;
069
070    /**
071     * The name of the compression type supported by this compressor.
072     */
073    protected String compressionType;
074
075    /**
076     * The value to be assigned to the TIFF <i>Compression</i> tag in the
077     * TIFF image metadata.
078     */
079    protected int compressionTagValue;
080
081    /**
082     * Whether the compression is lossless.
083     */
084    protected boolean isCompressionLossless;
085
086    /**
087     * The <code>ImageOutputStream</code> to be written.
088     */
089    protected ImageOutputStream stream;
090
091    /**
092     * Creates a compressor object for use in compressing TIFF data. This
093     * object may be passed to the
094     * {@link TIFFImageWriteParam#setTIFFCompressor(TIFFCompressor)}
095     * method to override the compressor of a supported compression type or
096     * to provide the implementation of the compression algorithm of an
097     * unsupported compression type.
098     *
099     * <p>The parameters <code>compressionTagValue</code> and
100     * <code>isCompressionLossless</code> are provided to accomodate
101     * compression types which are unknown. A compression type is
102     * "known" if it is either among those already supported by the
103     * TIFF writer (see {@link TIFFImageWriteParam}), or is listed in
104     * the TIFF 6.0 specification but not supported. If the compression
105     * type is unknown, the <code>compressionTagValue</code> and
106     * <code>isCompressionLossless</code> parameters are ignored.</p>
107     *
108     * @param compressionType The name of the compression type.
109     * @param compressionTagValue The value to be assigned to the TIFF
110     * <i>Compression</i> tag in the TIFF image metadata; ignored if
111     * <code>compressionType</code> is a known type.
112     * @param isCompressionLossless Whether the compression is lossless;
113     * ignored if <code>compressionType</code> is a known type.
114     *
115     * @throws IllegalArgumentException if <code>compressionType</code> is
116     * <code>null</code> or <code>compressionTagValue</code> is less than
117     * <code>1</code>.
118     */
119    public TIFFCompressor(String compressionType,
120                          int compressionTagValue,
121                          boolean isCompressionLossless) {
122        if(compressionType == null) {
123            throw new IllegalArgumentException("compressionType == null");
124        } else if(compressionTagValue < 1) {
125            throw new IllegalArgumentException("compressionTagValue < 1");
126        }
127
128        // Set the compression type.
129        this.compressionType = compressionType;
130
131        // Determine whether this type is either defined in the TIFF 6.0
132        // specification or is already supported.
133        int compressionIndex = -1;
134        String[] compressionTypes = TIFFImageWriter.compressionTypes;
135        int len = compressionTypes.length;
136        for(int i = 0; i < len; i++) {
137            if(compressionTypes[i].equals(compressionType)) {
138                // Save the index of the supported type.
139                compressionIndex = i;
140                break;
141            }
142        }
143
144        if(compressionIndex != -1) {
145            // Known compression type.
146            this.compressionTagValue =
147                TIFFImageWriter.compressionNumbers[compressionIndex];
148            this.isCompressionLossless =
149                TIFFImageWriter.isCompressionLossless[compressionIndex];
150        } else {
151            // Unknown compression type.
152            this.compressionTagValue = compressionTagValue;
153            this.isCompressionLossless = isCompressionLossless;
154        }
155    }
156
157    /**
158     * Retrieve the name of the compression type supported by this compressor.
159     *
160     * @return The compression type name.
161     */
162    public String getCompressionType() {
163        return compressionType;
164    }
165
166    /**
167     * Retrieve the value to be assigned to the TIFF <i>Compression</i> tag
168     * in the TIFF image metadata.
169     *
170     * @return The <i>Compression</i> tag value.
171     */
172    public int getCompressionTagValue() {
173        return compressionTagValue;
174    }
175
176    /**
177     * Retrieves a value indicating whether the compression is lossless.
178     *
179     * @return Whether the compression is lossless.
180     */
181    public boolean isCompressionLossless() {
182        return isCompressionLossless;
183    }
184
185    /**
186     * Sets the <code>ImageOutputStream</code> to be written.
187     *
188     * @param stream an <code>ImageOutputStream</code> to be written.
189     *
190     * @see #getStream
191     */
192    public void setStream(ImageOutputStream stream) {
193        this.stream = stream;
194    }
195
196    /**
197     * Returns the <code>ImageOutputStream</code> that will be written.
198     * 
199     * @return an <code>ImageOutputStream</code>.
200     *
201     * @see #setStream(ImageOutputStream)
202     */
203    public ImageOutputStream getStream() {
204        return stream;
205    }
206
207    /**
208     * Sets the value of the <code>writer</code> field.
209     *
210     * @param writer the current <code>ImageWriter</code>.
211     *
212     * @see #getWriter()
213     */
214    public void setWriter(ImageWriter writer) {
215        this.writer = writer;
216    }
217
218    /**
219     * Returns the current <code>ImageWriter</code>.
220     *
221     * @return an <code>ImageWriter</code>.
222     *
223     * @see #setWriter(ImageWriter)
224     */
225    public ImageWriter getWriter() {
226        return this.writer;
227    }
228
229    /**
230     * Sets the value of the <code>metadata</code> field.
231     *
232     * @param metadata the <code>IIOMetadata</code> object for the
233     * image being written.
234     *
235     * @see #getMetadata()
236     */
237    public void setMetadata(IIOMetadata metadata) {
238        this.metadata = metadata;
239    }
240
241    /**
242     * Returns the current <code>IIOMetadata</code> object.
243     *
244     * @return the <code>IIOMetadata</code> object for the image being
245     * written.
246     *
247     * @see #setMetadata(IIOMetadata)
248     */
249    public IIOMetadata getMetadata() {
250        return this.metadata;
251    }
252
253    /**
254     * Encodes the supplied image data, writing to the currently set
255     * <code>ImageOutputStream</code>.
256     *
257     * @param b an array of <code>byte</code>s containing the packed
258     * but uncompressed image data.
259     * @param off the starting offset of the data to be written in the
260     * array <code>b</code>.
261     * @param width the width of the rectangle of pixels to be written.
262     * @param height the height of the rectangle of pixels to be written.
263     * @param bitsPerSample an array of <code>int</code>s indicting
264     * the number of bits used to represent each image sample within
265     * a pixel.
266     * @param scanlineStride the number of bytes separating each
267     * row of the input data.
268     *
269     * @return the number of bytes written.
270     *
271     * @throws IOException if the supplied data cannot be encoded by
272     * this <code>TIFFCompressor</code>, or if any I/O error occurs
273     * during writing.
274     */
275    public abstract int encode(byte[] b, int off,
276                               int width, int height,
277                               int[] bitsPerSample,
278                               int scanlineStride) throws IOException;
279
280}
